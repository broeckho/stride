58 shared_ptr<Population> GeoPopBuilder::Build(shared_ptr<Population> pop)
59 {
60  m_stride_logger->trace("Building geopop.");
61 
62  // --------------------------------------------------------------
63  // Set the GeoGridConfig.
64  // --------------------------------------------------------------
65  GeoGridConfig ggConfig(m_config);
66  ggConfig.SetData(m_config.get<string>("run.geopop_gen.household_file"));
67 
68  // --------------------------------------------------------------
69  // Get GeoGrid associated with 'pop'.
70  // --------------------------------------------------------------
71  auto& geoGrid = pop->RefGeoGrid();
72 
73  // --------------------------------------------------------------
74  // Read cities input files (commute info file only if present).
75  // --------------------------------------------------------------
76  string commutesFile;
77  auto geopop_gen = m_config.get_child("run.geopop_gen");
78  if (geopop_gen.count("commuting_file")) {
79  commutesFile = m_config.get<string>("run.geopop_gen.commuting_file");
80  }
81 
82  m_stride_logger->trace("Starting MakeLocations");
83  MakeLocations(geoGrid, ggConfig, m_config.get<string>("run.geopop_gen.cities_file"), commutesFile);
84  m_stride_logger->trace("Finished MakeLocations");
85 
86  // --------------------------------------------------------------
87  // Generate Geo.
88  // --------------------------------------------------------------
89  m_stride_logger->trace("Starting MakeCenters");
90  MakeCenters(geoGrid, ggConfig);
91  m_stride_logger->trace("Finished MakeCenters");
92 
93  // --------------------------------------------------------------
94  // Generate Pop.
95  // --------------------------------------------------------------
96  m_stride_logger->trace("Starting MakePersons");
97  MakePersons(geoGrid, ggConfig);
98  m_stride_logger->trace("Finished MakePersons");
99 
100  // --------------------------------------------------------------
101  // Done.
102  // --------------------------------------------------------------
103  m_stride_logger->trace("Done building geopop.");
104 
105  return pop;
106 }

108 void GeoPopBuilder::MakeLocations(GeoGrid& geoGrid, const GeoGridConfig& geoGridConfig, const string& citiesFileName,
109  const string& commutingFileName)
110 {
111  const auto citiesReader = ReaderFactory::CreateCitiesReader(citiesFileName);
112  citiesReader->FillGeoGrid(geoGrid);
113 
114  if (!commutingFileName.empty()) {
115  const auto commutesReader = ReaderFactory::CreateCommutesReader(commutingFileName);
116  commutesReader->FillGeoGrid(geoGrid);
117  }
118 
119  for (const shared_ptr<Location>& loc : geoGrid) {
120  loc->SetPopCount(geoGridConfig.input.pop_size);
121  }
122  geoGrid.Finalize();
123 }

66 void Immunizer::Random(const SegmentedVector<ContactPool>& pools, vector<double>& immunityDistribution,
67  double immunityLinkProbability)
68 {
69  // Initialize a vector to count the population per age class [0-100].
70  vector<double> populationBrackets(100, 0.0);
71 
72  // Count individuals per age class and set all "susceptible" individuals "immune".
73  // note: focusing on measles, we expect the number of susceptible individuals
74  // to be less compared to the number of immune.
75  // TODO but this is a generic simulator
76  for (auto& c : pools) {
77  for (const auto& p : c.GetPool()) {
78  if (p->GetHealth().IsSusceptible()) {
79  p->GetHealth().SetImmune();
80  populationBrackets[p->GetAge()]++;
81  }
82  }
83  }
84 
85  // Sampler for int in [0, pools.size()) and for double in [0.0, 1.0).
86  const auto poolsSize = static_cast<int>(pools.size());
87  auto intGenerator = m_rn_man.GetUniformIntGenerator(0, poolsSize, 0U);
88  auto uniform01Generator = m_rn_man.GetUniform01Generator(0U);
89 
90  // Calculate the number of susceptible individuals per age class.
91  unsigned int numSusceptible = 0;
92  for (unsigned int age = 0; age < 100; age++) {
93  populationBrackets[age] = floor(populationBrackets[age] * (1 - immunityDistribution[age]));
94  numSusceptible += static_cast<unsigned int>(populationBrackets[age]);
95  }
96 
97  // Sample susceptible individuals, until all age-dependent quota are reached.
98  while (numSusceptible > 0) {
99  // random pool, random order of members
100  const ContactPool& p_pool = pools[intGenerator()];
101  const auto size = static_cast<unsigned int>(p_pool.GetPool().size());
102  vector<unsigned int> indices(size);
103  iota(indices.begin(), indices.end(), 0U);
104  m_rn_man.Shuffle(indices, 0U);
105 
106  // loop over members, in random order
107  for (unsigned int i_p = 0; i_p < size && numSusceptible > 0; i_p++) {
108  Person& p = *p_pool.GetPool()[indices[i_p]];
109  // if p is immune and his/her age class has not reached the quota => make susceptible
110  if (p.GetHealth().IsImmune() && populationBrackets[p.GetAge()] > 0) {
111  p.GetHealth().SetSusceptible();
112  populationBrackets[p.GetAge()]--;
113  numSusceptible--;
114  }
115  // random draw to continue in this pool or to sample a new one
116  if (uniform01Generator() < (1 - immunityLinkProbability)) {
117  break;
118  }
119  }
120  }
121 }

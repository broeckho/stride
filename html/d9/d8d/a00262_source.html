45 double Location::GetInfectedRatio() const
46 {
47  unsigned int infected = 0;
48  unsigned int population = 0;
49 
50  for (const std::shared_ptr<gengeopop::ContactCenter>& cc : m_contactCenters) {
51  auto r = cc->GetPopulationAndInfectedCount();
52  population += r.first;
53  infected += r.second;
54  }
55 
56  if (GetPopulation() == 0) {
57  return 0;
58  }
59 
60  double r = static_cast<double>(infected) / static_cast<double>(population);
61 
62  return r;
63 }

119 int Location::IncomingCommutingPeople(double fractionOfPopulationCommuting) const
120 {
121  double value = 0;
122  for (const auto& locProportion : m_incomingCommutingLocations) {
123  // locProportion.second of the people in locProportion.first are commuting to this
124  value += locProportion.second *
125  (fractionOfPopulationCommuting * (double)locProportion.first->GetPopulation());
126  }
127  return static_cast<int>(std::floor(value));
128 }

130 int Location::OutGoingCommutingPeople(double fractionOfPopulationCommuting) const
131 {
132  double totalProportion = 0;
133  for (const auto& locProportion : m_outgoingCommutingLocations) {
134  // locProportion.second of the people in this are commuting to locProportion.first
135  totalProportion += locProportion.second;
136  }
137  return static_cast<int>(
138  std::floor(totalProportion * (fractionOfPopulationCommuting * static_cast<double>(m_population))));

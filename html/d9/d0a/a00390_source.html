37 shared_ptr<Population> SurveySeeder::Seed(shared_ptr<Population> pop)
38 {
39  const string log_level = m_config_pt.get<string>("run.contact_log_level", "None");
40  if (log_level != "None") {
41  Population& population = *pop;
42  auto& logger = population.GetContactLogger();
43  const auto max_index = static_cast<unsigned int>(population.size() - 1);
44  auto generator = m_rn_manager[0].variate_generator(trng::uniform_int_dist(0, max_index));
45  const auto participants = m_config_pt.get<unsigned int>("run.num_participants_survey");
46 
47  // Use while-loop to get 'participants' unique participants (default sampling is with replacement).
48  // A for loop will not do because we might draw the same person twice.
49  auto num_samples = 0U;
50  while (num_samples < participants) {
51  Person& p = population[generator()];
52  if (!p.IsSurveyParticipant()) {
53  p.ParticipateInSurvey();
54  logger->info("[PART] {} {} {} {} {} {} {} {} {} {} {}", p.GetId(), p.GetAge(),
55  p.GetGender(), p.GetPoolId(Id::K12School), p.GetPoolId(Id::College),
56  p.GetPoolId(Id::Work), p.GetHealth().IsSusceptible(),
57  p.GetHealth().IsInfected(), p.GetHealth().IsInfectious(),
58  p.GetHealth().IsRecovered(), p.GetHealth().IsImmune());
59  num_samples++;
60  }
61  }
62  }
63  return pop;
64 }

107 void CliController::RegisterViewers(shared_ptr<SimRunner> runner)
108 {
109  // Command line viewer
110  m_stride_logger->info("Registering CliViewer");
111  const auto cli_v = make_shared<viewers::CliViewer>(runner, m_stride_logger);
112  runner->Register(cli_v, bind(&viewers::CliViewer::Update, cli_v, placeholders::_1));
113 
114  // Adopted viewer
115  if (m_config_pt.get<bool>("run.output_adopted", false)) {
116  m_stride_logger->info("registering AdoptedViewer,");
117  const auto v = make_shared<viewers::AdoptedViewer>(runner, m_output_prefix);
118  runner->Register(v, bind(&viewers::AdoptedViewer::Update, v, placeholders::_1));
119  }
120 
121  // Infection counts viewer
122  if (m_config_pt.get<bool>("run.output_cases", false)) {
123  m_stride_logger->info("Registering InfectedViewer");
124  const auto v = make_shared<viewers::InfectedViewer>(runner, m_output_prefix);
125  runner->Register(v, bind(&viewers::InfectedViewer::Update, v, placeholders::_1));
126  }
127 
128  // Persons viewer
129  if (m_config_pt.get<bool>("run.output_persons", false)) {
130  m_stride_logger->info("registering PersonsViewer.");
131  const auto v = make_shared<viewers::PersonsViewer>(runner, m_output_prefix);
132  runner->Register(v, bind(&viewers::PersonsViewer::Update, v, placeholders::_1));
133  }

135  // Summary viewer

137  m_stride_logger->info("Registering SummaryViewer");
138  const auto v = make_shared<viewers::SummaryViewer>(runner, m_output_prefix);
139  runner->Register(v, bind(&viewers::SummaryViewer::Update, v, placeholders::_1));
140  }

143 void CliController::LogSetup()
144 {
145  m_stride_logger->info("CliController stating up at: {}", TimeStamp().ToString());
146  m_stride_logger->info("Executing revision {}", ConfigInfo::GitRevision());
147  m_stride_logger->info("Creating dir: {}", m_output_prefix);
148  m_stride_logger->trace("Executing: {}", FileSys::GetExecPath().string());
149  m_stride_logger->trace("Current directory: {}", FileSys::GetCurrentDir().string());
150  if (m_use_install_dirs) {
151  m_stride_logger->trace("Install directory: {}", FileSys::GetRootDir().string());
152  m_stride_logger->trace("Config directory: {}", FileSys::GetConfigDir().string());
153  m_stride_logger->trace("Data directory: {}", FileSys::GetDataDir().string());
154  }
155  if (ConfigInfo::HaveOpenMP()) {
156  m_stride_logger->info("Max number OpenMP threads in this environment: {}",
157  ConfigInfo::NumberAvailableThreads());
158  m_stride_logger->info("Configured number of threads: {}",
159  m_config_pt.get<unsigned int>("run.num_threads"));
160  } else {
161  m_stride_logger->info("Not using OpenMP threads.");
162  }
163 }

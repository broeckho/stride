35 void PrimaryCommunityGenerator::Apply(GeoGrid& geoGrid, const GeoGridConfig& geoGridConfig)
36 {
37  // 1. calculate number of communities, each community has average 2000 persons
38  // 2. assign communities to a location using a discrete distribution reflecting the relative number of
39  // people at that location
40 
41  const auto popCount = geoGridConfig.input.pop_size;
42  const auto communitySize = geoGridConfig.pools.primary_community_size;
43  const auto communityCount = static_cast<unsigned int>(ceil(popCount / static_cast<double>(communitySize)));
44 
45  vector<double> weights;
46  for (const auto& loc : geoGrid) {
47  const auto weight = static_cast<double>(loc->GetPopCount()) / static_cast<double>(popCount);
48  AssertThrow(weight >= 0 && weight <= 1 && !std::isnan(weight),
49  "CommunityGenerator> Invalid weight: " + to_string(weight), m_logger);
50  weights.push_back(weight);
51  }
52 
53  if (weights.empty()) {
54  // trng can't handle empty vectors
55  return;
56  }
57 
58  const auto dist = m_rn_man.GetDiscreteGenerator(weights, 0U);
59  auto pop = geoGrid.GetPopulation();
60 
61  for (auto i = 0U; i < communityCount; i++) {
62  const auto loc = geoGrid[dist()];
63  AddPools(*loc, pop, geoGridConfig.pools.pools_per_primary_community);
64  }
65 }

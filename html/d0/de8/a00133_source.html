101 FileSys::Dirs FileSys::Initialize()
102 {
103  Dirs dirs;
104  //------- Retrieving path of executable
105  {
106 #if defined(WIN32)
107  char exePath[MAX_PATH];
108  HMODULE hModule = GetModuleHandle(NULL);
109  if (GetModuleFileName(NULL, exePath, sizeof(exePath)) != 0)
110  ;
111  {
112  dirs.m_exec_path = canonical(system_complete(exePath));
113  }
114 #elif defined(__linux__)
115  char exePath[PATH_MAX];
116  size_t size = ::readlink("/proc/self/exe", exePath, sizeof(exePath));
117  if (size > 0 && size < sizeof(exePath)) {
118  exePath[size] = '\0';
119  dirs.m_exec_path = canonical(system_complete(exePath));
120  }
121 #elif defined(__APPLE__)
122  char exePath[PATH_MAX];
123  uint32_t size = sizeof(exePath);
124  if (_NSGetExecutablePath(exePath, &size) == 0) {
125  dirs.m_exec_path = canonical(system_complete(exePath));
126  }
127 #endif
128  }
129 
130  //------- Retrieving root and bin directory (the subdirectory of the install root)
131  {
132  boost::filesystem::path exec_dir = dirs.m_exec_path.parent_path();
133  if (!dirs.m_exec_path.empty()) {
134 #if (__APPLE__)
135  if (exec_dir.filename().string() == "MacOS") {
136  // app
137  // -Contents <-Root Path
138  // -MacOS
139  // -executables
140  dirs.m_bin_dir = exec_dir;
141  dirs.m_root_dir = exec_dir.parent_path();
142  } else
143 #endif
144  if (ToLower(exec_dir.filename().string()) == "debug" ||
145  ToLower(exec_dir.filename().string()) == "release") {
146  // x/exec <-Root Path
147  // -bin
148  // -release/debug
149  // -executables
150  dirs.m_bin_dir = exec_dir.parent_path();
151  dirs.m_root_dir = exec_dir.parent_path().parent_path();
152  } else
153 #if (WIN32)
154  if (exec_dir.filename().string() != "bin") {
155  // Executables in root folder
156  dirs.m_bin_dir = exec_dir;
157  dirs.m_root_dir = exec_dir;
158  } else
159 #endif
160  {
161  // x/exec <-Root Path
162  // -bin
163  // -executables
164  dirs.m_bin_dir = exec_dir;
165  dirs.m_root_dir = exec_dir.parent_path();
166  }
167  }
168  }
169  //------- Current Dir
170  {
171  dirs.m_current_dir = system_complete(current_path());
172  }
173  //------- Config Dir
174  {
175  dirs.m_config_dir = dirs.m_root_dir / "config";
176  dirs.m_config_dir = is_directory(dirs.m_config_dir) ? dirs.m_config_dir : empty_path;
177  }
178  //------- Data Dir
179  {
180  dirs.m_data_dir = dirs.m_root_dir / "data";
181  dirs.m_data_dir = is_directory(dirs.m_data_dir) ? dirs.m_data_dir : empty_path;
182  }
183  //------- Tests Dir
184  {
185  dirs.m_tests_dir = dirs.m_root_dir / "tests";
186  dirs.m_tests_dir = is_directory(dirs.m_tests_dir) ? dirs.m_tests_dir : empty_path;
187  }
188 
189  return dirs;
190 }

198 ptree FileSys::ReadPtreeFile(const boost::filesystem::path& f_p)
199 {
200  ptree ret;
201  if (!exists(f_p) || !is_regular_file(f_p)) {
202  const string s = "FileSys::ReadPtreeFile> Abort! File " + f_p.string() + " not present.";
203  cerr << s << endl;
204  throw runtime_error(s);
205  } else {
206  try {
207  read_xml(canonical(f_p).string(), ret, xml_parser::trim_whitespace);
208  } catch (xml_parser_error& e) {
209  const string s = "FileSys::ReadPtreeFile> Abort! Error reading " + f_p.string();
210  cerr << s << endl;
211  throw runtime_error(s);
212  }

72 void Sim::TimeStep()
73 {
74  // Logic where you compute (on the basis of input/config for initial day or on the basis of
75  // number of sick persons, duration of epidemic etc) what kind of DaysOff scheme you apply.
76  const auto daysOff = std::make_shared<DaysOffStandard>(m_calendar);
77  const bool isWorkOff = daysOff->IsWorkOff();
78  const bool isSchoolOff = daysOff->IsSchoolOff();
79 
80  // To be used in update of population & contact pools.
81  Population& population = *m_population;
82  auto& poolSys = population.GetContactPoolSys();
83  auto contactLogger = population.GetContactLogger();
84  const auto simDay = m_calendar->GetSimulationDay();
85  const auto& infector = *m_infector;
86 
87 #pragma omp parallel num_threads(m_num_threads)
88  {
89  // Update health status and presence/absence in contact pools
90  // depending on health status, work/school day and whether
91  // we want to track index cases without adaptive behavior
92 #pragma omp for schedule(static)
93  for (size_t i = 0; i < population.size(); ++i) {
94  population[i].Update(isWorkOff, isSchoolOff, m_adaptive_symptomatic_behavior);
95  }
96 
97  // after the health update, let the public health agency perform their work
98  m_public_health_agency.Exec(m_population, m_rn_manager, simDay);
99 
100  // Infector updates individuals for contacts & transmission within each pool.
101  // Skip pools with id = 0, because it means Not Applicable.
102  const auto thread_num = static_cast<unsigned int>(omp_get_thread_num());
103  for (auto typ : ContactType::IdList) {
104  if ((typ == ContactType::Id::Workplace && isWorkOff) ||
105  (typ == ContactType::Id::K12School && isSchoolOff) ||
106  (typ == ContactType::Id::College && isSchoolOff)) {
107  continue;
108  }
109 #pragma omp for schedule(static)
110  for (size_t i = 1; i < poolSys[typ].size(); i++) { // NOLINT
111  infector(poolSys[typ][i], m_contact_profiles[typ], m_transmission_profile,
112  m_handlers[thread_num], simDay, contactLogger);
113  }
114  }
115  }
116 
117  m_population->GetContactLogger()->flush();
118  m_calendar->AdvanceDay();
119 }

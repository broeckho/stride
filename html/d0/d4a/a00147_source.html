85 template <typename T>
86 void RNManager::para_seed(vector<T>& engines, const Info& info)
87 {
88  auto state = info.m_state;
89  engines.resize(m_stream_count);
90  if (state.empty()) {
91  for (size_t i = 0; i < m_stream_count; ++i) {
92  engines[i].seed(m_seed);
93  engines[i].split(m_stream_count, i);
94  }
95  } else {
96  stringstream ss(state);
97  for (size_t i = 0; i < m_stream_count; ++i) {
98  ss >> engines[i];
99  engines[i].split(m_stream_count, i);
100  }
101  }
102 }

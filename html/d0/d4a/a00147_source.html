32 void CollegeGenerator::Apply(GeoGrid& geoGrid, const GeoGridConfig& geoGridConfig, unsigned int& ccCounter)
33 {
34  const auto studentCount = geoGridConfig.popInfo.popcount_college;
35  const auto collegeCount =
36  static_cast<unsigned int>(ceil(studentCount / static_cast<double>(geoGridConfig.pools.college_size)));
37  const auto cities = geoGrid.TopK(10);
38 
39  if (cities.empty()) {
40  // trng can't handle empty vectors
41  return;
42  }
43 
44  // Aggregate population in TopK cities.
45  auto totalPop = 0U;
46  for (const auto& c : cities) {
47  totalPop += c->GetPopCount();
48  }
49 
50  // Weights determined by relative population in city.
51  vector<double> weights;
52  for (const auto& c : cities) {
53  const auto weight = static_cast<double>(c->GetPopCount()) / static_cast<double>(totalPop);
54  AssertThrow(weight >= 0 && weight <= 1 && !std::isnan(weight),
55  "CollegeGenerator> Invalid weight: " + to_string(weight), m_logger);
56  weights.push_back(weight);
57  }
58 
59  const auto dist = m_rn_man.GetDiscreteGenerator(weights, 0U);
60  auto& poolSys = geoGrid.GetPopulation()->RefPoolSys();
61 
62  for (auto i = 0U; i < collegeCount; i++) {
63  auto loc = cities[dist()];
64  auto college = make_shared<ContactCenter>(ccCounter++, Id::College);
65 
66  for (auto j = 0U; j < geoGridConfig.pools.pools_per_college; ++j) {
67  const auto p = poolSys.CreateContactPool(Id::College);
68  college->RegisterPool(p);
69  loc->RegisterPool<Id::College>(p);
70  }
71 
72  loc->AddCenter(college);
73  }
74 }

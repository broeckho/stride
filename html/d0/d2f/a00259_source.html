28 void K12SchoolGenerator::Apply(shared_ptr<GeoGrid> geoGrid, const GeoGridConfig& geoGridConfig,
29  unsigned int& contactCenterCounter)
30 {
31  // 1. given the number of persons of school age, calculate number of schools; schools
32  // have 500 pupils on average
33  // 2. assign schools to a location by using a discrete distribution which reflects the
34  // relative number of pupils for that location; the relative number of pupils is set
35  // to the relative population w.r.t the total population.
36 
37  const auto pupilCount = geoGridConfig.popInfo.popcount_k12school;
38  const auto schoolCount =
39  static_cast<unsigned int>(ceil(pupilCount / static_cast<double>(geoGridConfig.pools.k12school_size)));
40 
41  vector<double> weights;
42  for (const auto& loc : *geoGrid) {
43  weights.push_back(loc->GetRelativePopulationSize());
44  }
45 
46  if (weights.empty()) {
47  // trng can't handle empty vectors
48  return;
49  }
50 
51  const auto dist = m_rn_man.GetDiscreteGenerator(weights, 0U);
52 
53  for (auto i = 0U; i < schoolCount; i++) {
54  const auto loc = (*geoGrid)[dist()];
55  const auto k12 = make_shared<K12School>(contactCenterCounter++);
56  k12->Fill(geoGridConfig, geoGrid);
57  loc->AddContactCenter(k12);
58  }
59 }

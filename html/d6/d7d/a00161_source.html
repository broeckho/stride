35 void CommunityGenerator::Apply(shared_ptr<GeoGrid> geoGrid, const GeoGridConfig& geoGridConfig,
36  unsigned int& contactCenterCounter)
37 {
38  // 1. calculate number of communities, each community has average 2000 persons
39  // 2. assign communities to a location using a discrete distribution reflecting the relative number of
40  // people at that location
41 
42  const auto popCount = geoGridConfig.input.pop_size;
43  const auto communityCount =
44  static_cast<unsigned int>(ceil(popCount / static_cast<double>(geoGridConfig.pools.community_size)));
45 
46  vector<double> weights;
47  for (const auto& loc : *geoGrid) {
48  const auto weight = static_cast<double>(loc->GetPopCount()) / static_cast<double>(popCount);
49  AssertThrow(weight >= 0 && weight <= 1 && !std::isnan(weight),
50  "CommunityGenerator> Invalid weight: " + to_string(weight), m_logger);
51  weights.push_back(weight);
52  }
53 
54  if (weights.empty()) {
55  // trng can't handle empty vectors
56  return;
57  }
58 
59  const auto dist = m_rnManager[0].variate_generator(trng::discrete_dist(weights.begin(), weights.end()));
60 
61  for (auto i = 0U; i < communityCount; i++) {
62  const auto loc = (*geoGrid)[dist()];
63  const auto pc = make_shared<PrimaryCommunity>(contactCenterCounter++);
64  pc->Fill(geoGridConfig, geoGrid);
65  loc->AddContactCenter(pc);
66  }
67  for (auto i = 0U; i < communityCount; i++) {
68  const auto loc = (*geoGrid)[dist()];
69  const auto sc = make_shared<SecondaryCommunity>(contactCenterCounter++);
70  sc->Fill(geoGridConfig, geoGrid);
71  loc->AddContactCenter(sc);
72  }
73 }

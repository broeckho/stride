29 void CommunityGenerator::Apply(shared_ptr<GeoGrid> geoGrid, GeoGridConfig& geoGridConfig)
30 {
31  // 1. calculate number of communities, each community has average 2000 persons
32  // 2. assign communities to a location using a discrete distribution reflecting the relative number of
33  // people at that location
34 
35  const auto popCount = geoGridConfig.input.populationSize;
36  const auto communityCount =
37  static_cast<unsigned int>(ceil(popCount / geoGridConfig.constants.meanCommunitySize));
38 
39  vector<double> weights;
40  for (const auto& loc : *geoGrid) {
41  const auto weight = static_cast<double>(loc->GetPopulation()) / static_cast<double>(popCount);
42  CheckWeight("CommunityGenerator", weight);
43  weights.push_back(weight);
44  }
45 
46  if (weights.empty()) {
47  // trng can't handle empty vectors
48  return;
49  }
50 
51  const auto dist = m_rnManager[0].variate_generator(trng::discrete_dist(weights.begin(), weights.end()));
52 
53  for (auto i = 0U; i < communityCount; i++) {
54  const auto loc = (*geoGrid)[dist()];
55  const auto pc = make_shared<PrimaryCommunity>(geoGridConfig.generated.contactCenters++);
56  pc->Fill(geoGrid);
57  loc->AddContactCenter(pc);
58  }
59  for (auto i = 0U; i < communityCount; i++) {
60  const auto loc = (*geoGrid)[dist()];
61  const auto sc = make_shared<SecondaryCommunity>(geoGridConfig.generated.contactCenters++);
62  sc->Fill(geoGrid);
63  loc->AddContactCenter(sc);
64  }
65 }

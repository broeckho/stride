42 shared_ptr<GeoGrid> GeoGridProtoReader::Read()
43 {
44  proto::GeoGrid protoGrid;
45  if (!protoGrid.ParseFromIstream(m_inputStream.get())) {
46  throw runtime_error("Failed to parse Proto file");
47  }
48  m_geoGrid = make_shared<GeoGrid>(m_population);
49 #pragma omp parallel
50 #pragma omp single
51  {
52  for (int idx = 0; idx < protoGrid.persons_size(); idx++) {
53  const proto::GeoGrid_Person& protoPerson = protoGrid.persons(idx);
54 #pragma omp task firstprivate(protoPerson)
55  {
56 #pragma omp critical
57  {
58  stride::Person* person = ParsePerson(protoPerson);
59  m_people[person->GetId()] = person;
60  }
61  }
62  }
63 #pragma omp taskwait
64  }
65  auto e = make_shared<ThreadException>();
66 #pragma omp parallel
67 #pragma omp single
68  {
69  for (int idx = 0; idx < protoGrid.locations_size(); idx++) {
70  shared_ptr<Location> loc;
71  const proto::GeoGrid_Location& protoLocation = protoGrid.locations(idx);
72 #pragma omp task firstprivate(protoLocation, loc)
73  {
74  e->Run([&loc, this, &protoLocation] { loc = ParseLocation(protoLocation); });
75  if (!e->HasError())
76 #pragma omp critical
77  m_geoGrid->AddLocation(move(loc));
78  }
79  }
80 #pragma omp taskwait
81  }
82  e->Rethrow();
83  AddCommutes(m_geoGrid);
84  m_people.clear();
85  m_commutes.clear();
86  return m_geoGrid;

89 shared_ptr<ContactCenter> GeoGridProtoReader::ParseContactCenter(
90  const proto::GeoGrid_Location_ContactCenter& protoContactCenter)
91 {
92  const auto type = protoContactCenter.type();
93  const auto id = protoContactCenter.id();
94 
95  shared_ptr<ContactCenter> result;
96  stride::ContactType::Id typeId;
97  switch (type) {
98  case proto::GeoGrid_Location_ContactCenter_Type_K12School:
99  result = make_shared<K12School>(id);
100  typeId = stride::ContactType::Id::K12School;
101  break;
102  case proto::GeoGrid_Location_ContactCenter_Type_PrimaryCommunity:
103  result = make_shared<PrimaryCommunity>(id);
104  typeId = stride::ContactType::Id::PrimaryCommunity;
105  break;
106  case proto::GeoGrid_Location_ContactCenter_Type_SecondaryCommunity:
107  result = make_shared<SecondaryCommunity>(id);
108  typeId = stride::ContactType::Id::SecondaryCommunity;
109  break;
110  case proto::GeoGrid_Location_ContactCenter_Type_College:
111  result = make_shared<College>(id);
112  typeId = stride::ContactType::Id::College;
113  break;
114  case proto::GeoGrid_Location_ContactCenter_Type_Household:
115  result = make_shared<Household>(id);
116  typeId = stride::ContactType::Id::Household;
117  break;
118  case proto::GeoGrid_Location_ContactCenter_Type_Workplace:
119  result = make_shared<Workplace>(id);
120  typeId = stride::ContactType::Id::Workplace;
121  break;
122  break;
123  default: throw runtime_error("No such ContactCenter type");
124  }
125 
126  auto e = make_shared<ThreadException>();
127 #pragma omp parallel
128 #pragma omp single
129  {
130  for (int idx = 0; idx < protoContactCenter.pools_size(); idx++) {
131  const proto::GeoGrid_Location_ContactCenter_ContactPool& protoContactPool =
132  protoContactCenter.pools(idx);
133 #pragma omp task firstprivate(protoContactPool, typeId)
134  {
135  stride::ContactPool* pool = nullptr;
136  e->Run([&protoContactPool, &pool, this, &typeId] {
137  pool = ParseContactPool(protoContactPool, typeId);
138  });
139  if (!e->HasError()) {
140 #pragma omp critical
141  result->RegisterPool(pool);
142  }
143  }
144  }
145 #pragma omp taskwait
146  }
147  e->Rethrow();
148 
149  return result;
150 }

157 stride::ContactPool* GeoGridProtoReader::ParseContactPool(
158  const proto::GeoGrid_Location_ContactCenter_ContactPool& protoContactPool, stride::ContactType::Id type)
159 {
160  // Don't use the id of the ContactPool but the let the Population create an id
161  stride::ContactPool* result;
162 
163 #pragma omp critical
164  result = m_geoGrid->CreateContactPool(type);
165 
166  for (int idx = 0; idx < protoContactPool.people_size(); idx++) {
167  const auto person_id = static_cast<unsigned int>(protoContactPool.people(idx));
168  const auto person = m_people.at(person_id);
169 
170 #pragma omp critical
171  {
172  result->AddMember(person);
173  // Update original pool id with new pool id used in the population
174  person->SetPoolId(type, static_cast<unsigned int>(result->GetId()));
175  }
176  }
177 
178  return result;
179 }

181 shared_ptr<Location> GeoGridProtoReader::ParseLocation(const proto::GeoGrid_Location& protoLocation)
182 {
183  const auto id = protoLocation.id();
184  const auto& name = protoLocation.name();
185  const auto province = protoLocation.province();
186  const auto population = protoLocation.population();
187  const auto& coordinate = ParseCoordinate(protoLocation.coordinate());
188 
189  auto result = make_shared<Location>(id, province, population, coordinate, name);
190 
191  auto e = make_shared<ThreadException>();
192 #pragma omp parallel
193 #pragma omp single
194  {
195  for (int idx = 0; idx < protoLocation.contactcenters_size(); idx++) {
196  shared_ptr<ContactCenter> center;
197  const proto::GeoGrid_Location_ContactCenter& protoCenter = protoLocation.contactcenters(idx);
198 #pragma omp task firstprivate(protoCenter, center)
199  {
200  e->Run([&protoCenter, this, &center] { center = ParseContactCenter(protoCenter); });
201  if (!e->HasError())
202 #pragma omp critical
203  result->AddContactCenter(center);
204  }
205  }
206 #pragma omp taskwait
207  }
208  e->Rethrow();
209 
210  for (int idx = 0; idx < protoLocation.commutes_size(); idx++) {
211  const proto::GeoGrid_Location_Commute& commute = protoLocation.commutes(idx);
212 #pragma omp critical
213  m_commutes.emplace_back(make_tuple(id, commute.to(), commute.proportion()));
214  }
215 
216  return result;
217 }

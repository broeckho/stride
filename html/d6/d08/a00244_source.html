30 void K12SchoolGenerator::Apply(GeoGrid& geoGrid, const GeoGridConfig& geoGridConfig, unsigned int& ccCounter)
31 {
32  // 1. given the number of persons of school age, calculate number of schools; schools
33  // have 500 pupils on average
34  // 2. assign schools to a location by using a discrete distribution which reflects the
35  // relative number of pupils for that location; the relative number of pupils is set
36  // to the relative population w.r.t the total population.
37 
38  const auto pupilCount = geoGridConfig.popInfo.popcount_k12school;
39  const auto schoolCount =
40  static_cast<unsigned int>(ceil(pupilCount / static_cast<double>(geoGridConfig.pools.k12school_size)));
41 
42  vector<double> weights;
43  for (const auto& loc : geoGrid) {
44  weights.push_back(loc->GetPopFraction());
45  }
46 
47  if (weights.empty()) {
48  // trng can't handle empty vectors
49  return;
50  }
51 
52  const auto dist = m_rn_man.GetDiscreteGenerator(weights, 0U);
53  auto& poolSys = geoGrid.GetPopulation()->RefPoolSys();
54 
55  for (auto i = 0U; i < schoolCount; i++) {
56  const auto loc = geoGrid[dist()];
57  const auto k12 = make_shared<ContactCenter>(ccCounter++, Id::K12School);
58 
59  for (auto j = 0U; j < geoGridConfig.pools.pools_per_k12school; ++j) {
60  const auto p = poolSys.CreateContactPool(Id::K12School);
61  k12->RegisterPool(p);
62  loc->RegisterPool<Id::K12School>(p);
63  }
64 
65  loc->AddCenter(k12);
66  }
67 }

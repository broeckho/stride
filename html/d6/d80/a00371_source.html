73 void Sim::TimeStep()
74 {
75  // Logic where you compute (on the basis of input/config for initial day or on the basis of
76  // number of sick persons, duration of epidemic etc) what kind of DaysOff scheme you apply.
77  const auto daysOff = std::make_shared<DaysOffStandard>(m_calendar);
78  const bool isWorkOff = daysOff->IsWorkOff();
79  const bool isSchoolOff = daysOff->IsSchoolOff();
80 
81  // To be used in update of population & contact pools.
82  Population& population = *m_population;
83  auto& poolSys = population.GetContactPoolSys();
84  auto contactLogger = population.GetContactLogger();
85  const auto simDay = m_calendar->GetSimulationDay();
86  const auto& infector = *m_infector;
87 
88 #pragma omp parallel num_threads(m_num_threads)
89  {
90  // Update health status and presence/absence in pools
91  // depending on health status, work/school day.
92 #pragma omp for schedule(static)
93  for (size_t i = 0; i < population.size(); ++i) {
94  population[i].Update(isWorkOff, isSchoolOff);
95  }
96 
97  // Infector updates individuals for contacts & transmission within each pool.
98  // Skip pools with id = 0, because it means Not Applicable.
99  const auto thread_num = static_cast<unsigned int>(omp_get_thread_num());
100  for (auto typ : ContactPoolType::IdList) {
101  if ((typ == ContactPoolType::Id::Work && isWorkOff) ||
102  (typ == ContactPoolType::Id::K12School && isSchoolOff) ||
103  (typ == ContactPoolType::Id::College && isSchoolOff)) {
104  continue;
105  }
106 #pragma omp for schedule(static)
107  for (size_t i = 1; i < poolSys[typ].size(); i++) { // NOLINT
108  infector(poolSys[typ][i], m_contact_profiles[typ], m_transmission_profile,
109  m_handlers[thread_num], simDay, contactLogger);
110  }
111  }
112  }
113 
114  m_population->GetContactLogger()->flush();
115  m_calendar->AdvanceDay();
116 }

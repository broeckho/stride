44 std::shared_ptr<Population> Population::Create(const boost::property_tree::ptree& configPt, util::RnMan& rnManager)
45 {
46  // --------------------------------------------------------------
47  // Create (empty) population & and give it a ContactLogger.
48  // --------------------------------------------------------------
49  struct make_shared_enabler : public Population

51  };
52  auto pop = make_shared<make_shared_enabler>();
53  if (configPt.get<bool>("run.contact_output_file", true)) {

58  } else {
59  pop->GetContactLogger() = LogUtils::CreateNullLogger("contact_logger");
60  }
61 
62  std::string geopop_type = configPt.get<std::string>("run.geopop_type", "default");
63 
64  if (geopop_type == "import") {
65  ImportPopBuilder(configPt, rnManager).Build(pop);
66  } else if (geopop_type == "generate") {
67  GenPopBuilder(configPt, rnManager).Build(pop);
68  } else {
69  DefaultPopBuilder(configPt, rnManager).Build(pop);
70  }
71  // -----------------------------------------------------------------------------------------
72  // Build population (at later date multiple builder or build instances ...).
73  // -----------------------------------------------------------------------------------------
74  return pop;

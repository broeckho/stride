64 int Location::GetIncomingCommuteCount(double fractionCommuters) const
65 {
66  double value = 0;
67  for (const auto& locProportion : m_inCommutes) {
68  // locProportion.second of the people in locProportion.first are commuting to this
69  value += locProportion.second * (fractionCommuters * (double)locProportion.first->GetPopCount());
70  }
71  return static_cast<int>(floor(value));
72 }

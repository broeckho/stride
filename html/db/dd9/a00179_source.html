131 void CSV::ReadFromStream(std::istream& inputStream)
132 {
133  std::string line;
134 
135  // header
136  getline(inputStream, line);
137  line = Trim(line);
138  std::vector<std::string> headerLabels = Split(line, ","); // Split is bad! There is no option to escape ",".
139  for (const std::string& label : headerLabels) {
140  m_labels.push_back(Trim(label, "\""));
141  }
142  m_column_count = m_labels.size();

144  // body

152  }

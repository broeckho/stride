194 vector<unsigned int> RunConfigManager::CreateNumThreads(unsigned int max)
195 {
196  max = std::max(max, ConfigInfo::NumberAvailableThreads());
197  initializer_list<unsigned int> num{1U};
198 
199  if (max >= 2) {
200  num = {1U, 2U};
201  }
202  if (max >= 4) {
203  num = {1U, 2U, 3U, 4U};
204  }
205  if (max >= 8) {
206  num = {1U, 2U, 3U, 4U, 5U, 6U, 7U, 8U};
207  }
208  if (max >= 12) {
209  num = {1U, 2U, 3U, 4U, 5U, 6U, 7U, 8U, 10U, 12U};
210  }
211  if (max >= 16) {
212  num = {1U, 2U, 3U, 4U, 5U, 6U, 7U, 8U, 10U, 12U, 16U};
213  }
214 
215  return num;
216 }

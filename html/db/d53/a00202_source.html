39 void GeoGridProtoReader::Read()
40 {
41  proto::GeoGrid protoGrid;
42  if (!protoGrid.ParseFromIstream(m_inputStream.get())) {
43  throw runtime_error("Failed to parse Proto file");
44  }
45  auto& geoGrid = m_population->RefGeoGrid();
46 
47 #pragma omp parallel
48 #pragma omp single
49  {
50  for (int idx = 0; idx < protoGrid.persons_size(); idx++) {
51  const proto::GeoGrid_Person& protoPerson = protoGrid.persons(idx);
52 #pragma omp task firstprivate(protoPerson)
53  {
54 #pragma omp critical
55  {
56  stride::Person* person = ParsePerson(protoPerson);
57  m_people[person->GetId()] = person;
58  }
59  }
60  }
61 #pragma omp taskwait
62  }
63  auto e = make_shared<ThreadException>();
64 #pragma omp parallel
65 #pragma omp single
66  {
67  for (int idx = 0; idx < protoGrid.locations_size(); idx++) {
68  shared_ptr<Location> loc;
69  const proto::GeoGrid_Location& protoLocation = protoGrid.locations(idx);
70 #pragma omp task firstprivate(protoLocation, loc)
71  {
72  e->Run([&loc, this, &protoLocation] { loc = ParseLocation(protoLocation); });
73  if (!e->HasError())
74 #pragma omp critical
75  geoGrid.AddLocation(move(loc));
76  }
77  }
78 #pragma omp taskwait
79  }
80  e->Rethrow();
81  AddCommutes(geoGrid);
82  m_people.clear();
83  m_commutes.clear();
84 }

86 shared_ptr<ContactCenter> GeoGridProtoReader::ParseContactCenter(
87  const proto::GeoGrid_Location_ContactCenter& protoContactCenter)
88 {
89  const auto type = protoContactCenter.type();
90  const auto id = protoContactCenter.id();
91 
92  Id typeId;
93  switch (type) {
94  case proto::GeoGrid_Location_ContactCenter_Type_K12School: typeId = Id::K12School; break;
95  case proto::GeoGrid_Location_ContactCenter_Type_PrimaryCommunity: typeId = Id::PrimaryCommunity; break;
96  case proto::GeoGrid_Location_ContactCenter_Type_SecondaryCommunity: typeId = Id::SecondaryCommunity; break;
97  case proto::GeoGrid_Location_ContactCenter_Type_College: typeId = Id::College; break;
98  case proto::GeoGrid_Location_ContactCenter_Type_Household: typeId = Id::Household; break;
99  case proto::GeoGrid_Location_ContactCenter_Type_Workplace: typeId = Id::Workplace; break;
100  default: throw runtime_error("No such ContactCenter type");
101  }
102  auto result = make_shared<ContactCenter>(id, typeId);
103 
104  auto e = make_shared<ThreadException>();
105 #pragma omp parallel
106 #pragma omp single
107  {
108  for (int idx = 0; idx < protoContactCenter.pools_size(); idx++) {
109  const proto::GeoGrid_Location_ContactCenter_ContactPool& protoContactPool =
110  protoContactCenter.pools(idx);
111 #pragma omp task firstprivate(protoContactPool, typeId)
112  {
113  stride::ContactPool* pool = nullptr;
114  e->Run([&protoContactPool, &pool, this, &typeId] {
115  pool = ParseContactPool(protoContactPool, typeId);
116  });
117  if (!e->HasError()) {
118 #pragma omp critical
119  result->RegisterPool(pool);
120  }
121  }
122  }
123 #pragma omp taskwait
124  }
125  e->Rethrow();
126 
127  return result;
128 }

135 stride::ContactPool* GeoGridProtoReader::ParseContactPool(
136  const proto::GeoGrid_Location_ContactCenter_ContactPool& protoContactPool, Id type)
137 {
138  // Don't use the id of the ContactPool but the let the Population create an id
139  stride::ContactPool* result;
140 
141 #pragma omp critical
142  result = m_population->RefPoolSys().CreateContactPool(type);
143 
144  for (int idx = 0; idx < protoContactPool.people_size(); idx++) {
145  const auto person_id = static_cast<unsigned int>(protoContactPool.people(idx));
146  const auto person = m_people.at(person_id);
147 
148 #pragma omp critical
149  {
150  result->AddMember(person);
151  // Update original pool id with new pool id used in the population
152  person->SetPoolId(type, static_cast<unsigned int>(result->GetId()));
153  }
154  }
155 
156  return result;
157 }

159 shared_ptr<Location> GeoGridProtoReader::ParseLocation(const proto::GeoGrid_Location& protoLocation)
160 {
161  const auto id = protoLocation.id();
162  const auto& name = protoLocation.name();
163  const auto province = protoLocation.province();
164  const auto population = protoLocation.population();
165  const auto& coordinate = ParseCoordinate(protoLocation.coordinate());
166 
167  auto result = make_shared<Location>(id, province, coordinate, name, population);
168 
169  auto e = make_shared<ThreadException>();
170 #pragma omp parallel
171 #pragma omp single
172  {
173  for (int idx = 0; idx < protoLocation.contactcenters_size(); idx++) {
174  shared_ptr<ContactCenter> center;
175  const proto::GeoGrid_Location_ContactCenter& protoCenter = protoLocation.contactcenters(idx);
176 #pragma omp task firstprivate(protoCenter, center)
177  {
178  e->Run([&protoCenter, this, &center] { center = ParseContactCenter(protoCenter); });
179  if (!e->HasError())
180 #pragma omp critical
181  result->AddCenter(center);
182  }
183  }
184 #pragma omp taskwait
185  }
186  e->Rethrow();
187 
188  for (int idx = 0; idx < protoLocation.commutes_size(); idx++) {
189  const proto::GeoGrid_Location_Commute& commute = protoLocation.commutes(idx);
190 #pragma omp critical
191  m_commutes.emplace_back(make_tuple(id, commute.to(), commute.proportion()));
192  }
193 
194  return result;
195 }

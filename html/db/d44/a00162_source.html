33 void CommutesCSVReader::FillGeoGrid(shared_ptr<GeoGrid> geoGrid) const
34 {
35  // flanders_commuting format
36  // kolom: stad van vertrek (headers = id)
37  // rij: stad van aankomst (volgorde = volgorde van kolommen = id).
38 
39  // represents the location id for column x
40  vector<unsigned int> header;
41 
42  for (const string& label : m_reader.GetLabels()) {
43  header.push_back(static_cast<unsigned int>(stoi(label.substr(3))));
44  }
45 
46  const size_t columnCount = m_reader.GetColumnCount();
47 
48  map<unsigned int, unsigned int> sizes; // indexed by header/row id
49 
50  // Since columns represent the "from city" and the proportion is calculated using the from city,
51  // the total population of a city is calculated using the values found in the columns.
52  for (const CSVRow& row : m_reader) {
53  for (size_t columnIndex = 0; columnIndex < columnCount; columnIndex++) {
54  sizes[columnIndex] += row.GetValue<int>(columnIndex);
55  }
56  }
57 
58  size_t rowIndex = 0;
59  for (const CSVRow& row : m_reader) {
60  for (size_t columnIndex = 0; columnIndex < columnCount; columnIndex++) {
61  auto abs = row.GetValue<double>(columnIndex);
62  if (abs != 0 && columnIndex != rowIndex) {
63  const auto& locFrom = geoGrid->GetById(header[columnIndex]);
64  const auto& locTo = geoGrid->GetById(header[rowIndex]);
65  const auto& total = sizes[columnIndex];
66  double proportion = abs / total;
67 
68  if (proportion < 0 || proportion > 1) {
69  throw Exception("Proportion of commutes from " + to_string(locFrom->GetID()) +
70  " to " + to_string(locTo->GetID()) +
71  " is invalid (0 <= proportion <= 1)");
72  }
73  locFrom->AddOutgoingCommutingLocation(locTo, proportion);
74  locTo->AddIncomingCommutingLocation(locFrom, proportion);
75  }
76  }
77  rowIndex++;
78  }
79 }

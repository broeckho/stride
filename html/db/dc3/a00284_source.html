51 std::shared_ptr<Population> Population::Create(const boost::property_tree::ptree& configPt, util::RnMan& rnManager,
52  std::shared_ptr<spdlog::logger> stride_logger)
53 {
54  if (!stride_logger) {
55  stride_logger = LogUtils::CreateNullLogger("Population_logger");
56  }
57 
58  // --------------------------------------------------------------
59  // Create (empty) population & and give it a ContactLogger.
60  // --------------------------------------------------------------
61  const auto pop = Create();
62  if (configPt.get<bool>("run.contact_output_file", true)) {
63  const auto prefix = configPt.get<string>("run.output_prefix");
64  const auto logPath = FileSys::BuildPath(prefix, "contact_log.txt");
65  pop->GetContactLogger() = LogUtils::CreateRotatingLogger("contact_logger", logPath.string());
66  pop->GetContactLogger()->set_pattern("%v");
67  stride_logger->info("Contact logging requested; logger set up.");
68  } else {
69  pop->GetContactLogger() = LogUtils::CreateNullLogger("contact_logger");
70  stride_logger->info("No contact logging requested.");
71  }
72 
73  // -----------------------------------------------------------------------------------------
74  // Build population.
75  // -----------------------------------------------------------------------------------------
76  std::string pop_type = configPt.get<std::string>("run.population_type", "default");
77  if (pop_type == "import") {
78  stride_logger->info("ImportPopBuilder invoked.");
79  ImportPopBuilder(configPt, rnManager).Build(pop);
80  } else if (pop_type == "generate") {
81  stride_logger->info("GeoPopBuilder invoked.");
82  GeoPopBuilder(configPt, rnManager).Build(pop);
83  } else {
84  stride_logger->info("DefaultPopBuilder invoked.");
85  DefaultPopBuilder(configPt, rnManager).Build(pop);
86  }
87 
88  // -----------------------------------------------------------------------------------------
89  // Done.
90  // -----------------------------------------------------------------------------------------
91  return pop;
92 }

42 std::shared_ptr<Population> Population::Create(const boost::property_tree::ptree& configPt, util::RnMan& rnManager,
43  std::shared_ptr<spdlog::logger> stride_logger)
44 {
45  if (!stride_logger) {
46  stride_logger = LogUtils::CreateNullLogger("Population_logger");
47  }
48 
49  // --------------------------------------------------------------
50  // Create (empty) population & and give it a ContactLogger.
51  // --------------------------------------------------------------
52  const auto pop = Create();
53  if (configPt.get<bool>("run.contact_output_file", true)) {
54  const auto prefix = configPt.get<string>("run.output_prefix");
55  const auto logPath = FileSys::BuildPath(prefix, "contact_log.txt");
56  pop->GetContactLogger() = LogUtils::CreateRotatingLogger("contact_logger", logPath.string());
57  pop->GetContactLogger()->set_pattern("%v");
58  stride_logger->info("Contact logging requested; logger set up.");
59  } else {
60  pop->GetContactLogger() = LogUtils::CreateNullLogger("contact_logger");
61  stride_logger->info("No contact logging requested.");
62  }
63 
64  // -----------------------------------------------------------------------------------------
65  // Build population.
66  // -----------------------------------------------------------------------------------------
67  std::string pop_type = configPt.get<std::string>("run.population_type", "default");
68  if (pop_type == "import") {
69  stride_logger->info("ImportPopBuilder invoked.");
70  ImportPopBuilder(configPt, rnManager).Build(pop);
71  } else if (pop_type == "generate") {
72  stride_logger->info("GenPopBuilder invoked.");
73  GeoPopBuilder(configPt, rnManager).Build(pop);
74  } else {
75  stride_logger->info("DefaultPopBuilder invoked.");
76  DefaultPopBuilder(configPt, rnManager).Build(pop);
77  }
78 
79  // -----------------------------------------------------------------------------------------
80  // Done.
81  // -----------------------------------------------------------------------------------------
82  return pop;
83 }

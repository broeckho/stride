45 shared_ptr<GeoGrid> GeoGridJSONReader::Read()
46 {
47  boost::property_tree::ptree root;
48  try {
49  boost::property_tree::read_json(*m_inputStream, root);
50  } catch (runtime_error&) {
51  throw Exception("Problem parsing JSON file, check whether empty or invalid JSON.");
52  }
53  m_geoGrid = make_shared<GeoGrid>(m_population);
54  auto people = root.get_child("persons");
55 #pragma omp parallel
56 #pragma omp single
57  {
58  for (auto it = people.begin(); it != people.end(); it++) {
59 #pragma omp task firstprivate(it)
60  {
61 #pragma omp critical
62  {
63  auto person = ParsePerson(it->second.get_child(""));
64  m_people[person->GetId()] = person;
65  }
66  }
67  }
68 #pragma omp taskwait
69  }
70  auto locations = root.get_child("locations");
71  auto e = make_shared<ThreadException>();
72 #pragma omp parallel
73 #pragma omp single
74  {
75  for (auto it = locations.begin(); it != locations.end(); it++) {
76  shared_ptr<Location> loc;
77 #pragma omp task firstprivate(it, loc)
78  {
79  e->Run([&loc, this, &it] { loc = ParseLocation(it->second.get_child("")); });
80  if (!e->HasError())
81 #pragma omp critical
82  m_geoGrid->AddLocation(move(loc));
83  }
84  }
85 #pragma omp taskwait
86  }
87  e->Rethrow();
88  AddCommutes(m_geoGrid);
89  m_commutes.clear();
90  m_people.clear();
91  return m_geoGrid;
92 }

94 shared_ptr<Location> GeoGridJSONReader::ParseLocation(boost::property_tree::ptree& location)
95 {
96  const auto id = boost::lexical_cast<unsigned int>(location.get<string>("id"));
97  const auto name = location.get<string>("name");
98  const auto province = boost::lexical_cast<unsigned int>(location.get<string>("province"));
99  const auto population = boost::lexical_cast<unsigned int>(location.get<string>("population"));
100  const auto coordinate = ParseCoordinate(location.get_child("coordinate"));
101 
102  auto result = make_shared<Location>(id, province, population, coordinate, name);
103  auto contactCenters = location.get_child("contactCenters");
104  auto e = make_shared<ThreadException>();
105 
106 #pragma omp parallel
107 #pragma omp single
108  {
109  for (auto it = contactCenters.begin(); it != contactCenters.end(); it++) {
110  shared_ptr<ContactCenter> center;
111 #pragma omp task firstprivate(it, center)
112  {
113  e->Run([&it, this, &center] { center = ParseContactCenter(it->second.get_child("")); });
114  if (!e->HasError())
115 #pragma omp critical
116  result->AddContactCenter(center);
117  }
118  }
119 #pragma omp taskwait
120  }
121  e->Rethrow();
122 
123  if (location.count("commutes")) {
124  boost::property_tree::ptree commutes = location.get_child("commutes");
125  for (auto it = commutes.begin(); it != commutes.end(); it++) {
126  const auto to = boost::lexical_cast<unsigned int>(it->first);
127  const auto amount = boost::lexical_cast<double>(it->second.data());
128 #pragma omp critical
129  m_commutes.emplace_back(id, to, amount);
130  }
131  }
132 
133  return result;
134 }

143 shared_ptr<ContactCenter> GeoGridJSONReader::ParseContactCenter(boost::property_tree::ptree& contactCenter)
144 {
145  const auto type = contactCenter.get<string>("type");
146  const auto id = boost::lexical_cast<unsigned int>(contactCenter.get<string>("id"));
147 
148  shared_ptr<ContactCenter> result;
149  ContactType::Id typeId;
150  if (type == ToString(Id::K12School)) {
151  result = make_shared<K12School>(id);
152  typeId = Id::K12School;
153  } else if (type == ToString(Id::College)) {
154  result = make_shared<College>(id);
155  typeId = ContactType::Id::College;
156  } else if (type == ToString(Id::Household)) {
157  result = make_shared<Household>(id);
158  typeId = ContactType::Id::Household;
159  } else if (type == ToString(Id::PrimaryCommunity)) {
160  result = make_shared<PrimaryCommunity>(id);
161  typeId = ContactType::Id::PrimaryCommunity;
162  } else if (type == ToString(Id::SecondaryCommunity)) {
163  result = make_shared<SecondaryCommunity>(id);
164  typeId = ContactType::Id::SecondaryCommunity;
165  } else if (type == ToString(Id::Workplace)) {
166  result = make_shared<Workplace>(id);
167  typeId = ContactType::Id::Workplace;
168  } else {
169  throw Exception("No such ContactCenter type: " + type);
170  }
171 
172  auto contactPools = contactCenter.get_child("pools");
173  auto e = make_shared<ThreadException>();
174 
175 #pragma omp parallel
176 #pragma omp single
177  {
178  for (auto it = contactPools.begin(); it != contactPools.end(); it++) {
179 #pragma omp task firstprivate(it)
180  {
181  ContactPool* pool = nullptr;
182  e->Run([&it, &pool, this, typeId] {
183  pool = ParseContactPool(it->second.get_child(""), typeId);
184  });
185  if (!e->HasError()) {
186 #pragma omp critical
187  result->RegisterPool(pool);
188  }
189  }
190  }
191 #pragma omp taskwait
192  }
193  e->Rethrow();
194  return result;
195 }

197 ContactPool* GeoGridJSONReader::ParseContactPool(boost::property_tree::ptree& contactPool, ContactType::Id typeId)
198 {
199  // Don't use the id of the ContactPool but the let the Population create an id.
200  auto result = m_geoGrid->CreateContactPool(typeId);
201  auto people = contactPool.get_child("people");
202 
203  for (auto it = people.begin(); it != people.end(); it++) {
204  auto person_id = boost::lexical_cast<unsigned int>(it->second.get<string>(""));
205  if (m_people.count(person_id) == 0) {
206  throw Exception("No such person: " + to_string(person_id));
207  }
208 #pragma omp critical
209  result->AddMember(m_people[person_id]);
210  }
211 
212  return result;
213 }

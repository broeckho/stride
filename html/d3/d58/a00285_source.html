42 bool Populator::MakeChoice(double fraction)
43 {
44  vector<double> weights;
45  weights.push_back(1.0 - fraction); // -> 0, return is false -> not part of the fraction
46  weights.push_back(fraction); // -> 1, return is true -> part of the fraction
47 
48  auto dist = m_rnManager[0].variate_generator(trng::discrete_dist(weights.begin(), weights.end()));
49  return static_cast<bool>(dist());
50 }

52 template <typename T>
53 vector<ContactPool*> Populator::GetNearbyPools(const shared_ptr<GeoGrid>& geoGrid, const shared_ptr<Location>& start,
54  double startRadius) const
55 {
56  double currentRadius = startRadius;
57  vector<ContactPool*> pools;
58 
59  while (pools.empty()) {
60  for (const shared_ptr<Location>& nearLoc : geoGrid->LocationsInRadius(start, currentRadius)) {
61  const auto& centers = nearLoc->GetContactCentersOfType<T>();
62  for (const auto& center : centers) {
63  pools.insert(pools.end(), center->begin(), center->end());
64  }
65  }
66  currentRadius *= 2;
67  if (currentRadius == numeric_limits<double>::infinity()) {
68  break;
69  }
70  }
71  return pools;
72 }

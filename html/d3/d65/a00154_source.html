43 shared_ptr<Sim> SimBuilder::Build(shared_ptr<Sim> sim, shared_ptr<Population> pop)
44 {
45  // --------------------------------------------------------------
46  // Read config info and setup random number manager
47  // --------------------------------------------------------------
48  sim->m_config_pt = m_config_pt;
49  sim->m_population = std::move(pop);
50  sim->m_track_index_case = m_config_pt.get<bool>("run.track_index_case");
51  sim->m_num_threads = m_config_pt.get<unsigned int>("run.num_threads");
52  sim->m_calendar = make_shared<Calendar>(m_config_pt);
53  sim->m_local_info_policy = m_config_pt.get<string>("run.local_information_policy", "NoLocalInformation");
54  sim->m_contact_log_mode = ContactLogMode::ToMode(m_config_pt.get<string>("run.contact_log_level", "None"));
55  sim->m_rn_manager.Initialize(RNManager::Info{m_config_pt.get<string>("run.rng_type", "mrg2"),
56  m_config_pt.get<unsigned long>("run.rng_seed", 1UL), "",
57  sim->m_num_threads});
58 
59  // --------------------------------------------------------------
60  // Contact handlers, each with generator bound to different
61  // random engine stream) and infector.
62  // --------------------------------------------------------------
63  for (size_t i = 0; i < sim->m_num_threads; i++) {
64  auto gen = sim->m_rn_manager.GetGenerator(trng::uniform01_dist<double>(), i);
65  sim->m_handlers.emplace_back(ContactHandler(gen));
66  }
67  const auto& select = make_tuple(sim->m_contact_log_mode, sim->m_track_index_case, sim->m_local_info_policy);
68  sim->m_infector = InfectorMap().at(select);
69 
70  // --------------------------------------------------------------
71  // Initialize the age-related contact profiles.
72  // --------------------------------------------------------------
73  const auto ageContactPt = ReadAgeContactPtree();
74  for (Id typ : IdList) {
75  sim->m_contact_profiles[typ] = AgeContactProfile(typ, ageContactPt);
76  }
77 
78  // --------------------------------------------------------------
79  // Initialize the transmission profile (fixes rates).
80  // --------------------------------------------------------------
81  const auto diseasePt = ReadDiseasePtree();
82  sim->m_transmission_profile.Initialize(m_config_pt, diseasePt);
83 
84  // --------------------------------------------------------------
85  // Seed the population with health data.
86  // --------------------------------------------------------------
87  HealthSeeder(diseasePt).Seed(sim->m_population, sim->m_handlers);
88 
89  // --------------------------------------------------------------
90  // Seed population wrt immunity/vaccination/infection.
91  // --------------------------------------------------------------
92  DiseaseSeeder(m_config_pt, sim->m_rn_manager).Seed(sim->m_population);
93 
94  // --------------------------------------------------------------
95  // Seed population wrt belief policies.
96  // --------------------------------------------------------------
97  BeliefSeeder(m_config_pt, sim->m_rn_manager).Seed(sim->m_population);
98 
99  // --------------------------------------------------------------
100  // Done.
101  // --------------------------------------------------------------
102  return sim;
103 }

30 void CollegePopulator::Apply(shared_ptr<GeoGrid> geoGrid, GeoGridConfig& geoGridConfig)
31 {
32  m_logger->info("Starting to populate Colleges");
33 
34  set<ContactPool*> found;
35  auto students = 0U;
36  auto commuting = 0U;
37 
38  // for every location
39  for (const shared_ptr<Location>& loc : *geoGrid) {
40  if (loc->GetPopulation() == 0) {
41  continue;
42  }
43  // 1. find all highschools in an area of 10-k*10 km
44  const vector<ContactPool*>& nearByColleges =
45  GetContactPoolInIncreasingRadius<College>(geoGrid, loc);
46 
47  ExcAssert(!nearByColleges.empty(), "No HighSchool found due to invalid input in CollegePopulator");
48 
49  auto distNonCommuting = m_rnManager[0].variate_generator(trng::uniform_int_dist(
50  0, static_cast<trng::uniform_int_dist::result_type>(nearByColleges.size())));
51 
52  // 2. find all highschools were students from this location commute to
53  vector<Location*> commutingHighSchools;
54  vector<double> commutingWeights;
55  for (const pair<Location*, double>& commute : loc->GetOutgoingCommuningCities()) {
56  const auto& highSchools = commute.first->GetContactCentersOfType<College>();
57  if (!highSchools.empty()) {
58  commutingHighSchools.push_back(commute.first);
59  commutingWeights.push_back(commute.second);
60  }
61  }
62 
63  function<trng::discrete_dist::result_type()> disCommuting;
64 
65  if (!commutingWeights.empty()) {
66  disCommuting = m_rnManager[0].variate_generator(
67  trng::discrete_dist(commutingWeights.begin(), commutingWeights.end()));
68  }
69 
70  // 2. for every student assign a class
71  for (const shared_ptr<ContactCenter>& household : loc->GetContactCentersOfType<Household>()) {
72  ContactPool* contactPool = household->GetPools()[0];
73  found.insert(contactPool);
74  for (Person* p : *contactPool) {
75  if (p->IsCollegeStudentCandidate() &&
76  MakeChoice(geoGridConfig.input.fraction_1826_years_WhichAreStudents)) {
77  students++;
78  // this person is a student
79  if (!commutingHighSchools.empty() &&
80  MakeChoice(geoGridConfig.input.fraction_student_commutingPeople)) {
81  // this person is commuting
82  commuting++;
83 
84  // id of the location this person is commuting to
85  auto locationId = disCommuting();
86  // create list of classes for each highschool at this location
87  const auto& highSchools = commutingHighSchools[locationId]
88  ->GetContactCentersOfType<College>();
89 
90  vector<ContactPool*> contactPools;
91  for (const auto& highSchool : highSchools) {
92  contactPools.insert(contactPools.end(), highSchool->begin(),
93  highSchool->end());
94  }
95 
96  auto disPools = m_rnManager[0].variate_generator(trng::uniform_int_dist(
97  0, static_cast<trng::uniform_int_dist::result_type>(
98  contactPools.size())));
99 
100  auto id = disPools();
101  contactPools[id]->AddMember(p);
102  p->SetCollegeId(contactPools[id]->GetId());
103  } else {
104  auto id = distNonCommuting();
105  nearByColleges[id]->AddMember(p);
106  p->SetCollegeId(
107  static_cast<unsigned int>(nearByColleges[id]->GetId()));
108  }
109  }
110  }
111  }
112  }
113  m_logger->info("Number of students in Colleges: {}", students);
114  m_logger->info("Number of classes: {}", found.size());
115  m_logger->info("Number students that commute: ", commuting);
116 }

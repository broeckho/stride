95 template <>
96 void Rn<pcg64>::Seed(randutils::seed_seq_fe128& seseq)
97 {
98  if (2 * m_stream_count > 64) {
99  throw std::runtime_error("RnPcg64 generate seed vector, cannot handle large n.");
100  }
101  auto seeds = pcg_extras::generate_vector<pcg64::state_type, 64>(seseq);
102  for (size_t i = 0; i < m_stream_count; ++i) {
103  (*this)[i].engine().seed(seeds[i + 1], seeds[i]);

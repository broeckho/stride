42 shared_ptr<GeoGrid> GeoGridProtoReader::Read()
43 {
44  proto::GeoGrid protoGrid;
45  if (!protoGrid.ParseFromIstream(m_inputStream.get())) {
46  throw runtime_error("Failed to parse Proto file");
47  }
48  m_geoGrid = make_shared<GeoGrid>(m_population);
49 #pragma omp parallel
50 #pragma omp single
51  {
52  for (int idx = 0; idx < protoGrid.persons_size(); idx++) {
53  const proto::GeoGrid_Person& protoPerson = protoGrid.persons(idx);
54 #pragma omp task firstprivate(protoPerson)
55  {
56 #pragma omp critical
57  {
58  stride::Person* person = ParsePerson(protoPerson);
59  m_people[person->GetId()] = person;
60  }
61  }
62  }
63 #pragma omp taskwait
64  }
65  auto e = make_shared<ThreadException>();
66 #pragma omp parallel
67 #pragma omp single
68  {
69  for (int idx = 0; idx < protoGrid.locations_size(); idx++) {
70  shared_ptr<Location> loc;
71  const proto::GeoGrid_Location& protoLocation = protoGrid.locations(idx);
72 #pragma omp task firstprivate(protoLocation, loc)
73  {
74  e->Run([&loc, this, &protoLocation] { loc = ParseLocation(protoLocation); });
75  if (!e->HasError())
76 #pragma omp critical
77  m_geoGrid->AddLocation(move(loc));
78  }
79  }
80 #pragma omp taskwait
81  }
82  e->Rethrow();
83  AddCommutes(m_geoGrid);
84  m_people.clear();
85  m_commutes.clear();
86  return m_geoGrid;
87 }

89 shared_ptr<Location> GeoGridProtoReader::ParseLocation(const proto::GeoGrid_Location& protoLocation)
90 {
91  const auto id = protoLocation.id();
92  const auto& name = protoLocation.name();
93  const auto province = protoLocation.province();
94  const auto population = protoLocation.population();
95  const auto& coordinate = ParseCoordinate(protoLocation.coordinate());
96 
97  auto result = make_shared<Location>(id, province, population, coordinate, name);
98 
99  auto e = make_shared<ThreadException>();
100 #pragma omp parallel
101 #pragma omp single
102  {
103  for (int idx = 0; idx < protoLocation.contactcenters_size(); idx++) {
104  shared_ptr<ContactCenter> center;
105  const proto::GeoGrid_Location_ContactCenter& protoCenter = protoLocation.contactcenters(idx);
106 #pragma omp task firstprivate(protoCenter, center)
107  {
108  e->Run([&protoCenter, this, &center] { center = ParseContactCenter(protoCenter); });
109  if (!e->HasError())
110 #pragma omp critical
111  result->AddContactCenter(center);
112  }
113  }
114 #pragma omp taskwait
115  }
116  e->Rethrow();
117 
118  for (int idx = 0; idx < protoLocation.commutes_size(); idx++) {
119  const proto::GeoGrid_Location_Commute& commute = protoLocation.commutes(idx);
120 #pragma omp critical
121  m_commutes.emplace_back(make_tuple(id, commute.to(), commute.proportion()));
122  }
123 
124  return result;
125 }

132 shared_ptr<ContactCenter> GeoGridProtoReader::ParseContactCenter(
133  const proto::GeoGrid_Location_ContactCenter& protoContactCenter)
134 {
135  const auto type = protoContactCenter.type();
136  const auto id = protoContactCenter.id();
137 
138  shared_ptr<ContactCenter> result;
139  stride::ContactType::Id typeId;
140  switch (type) {
141  case proto::GeoGrid_Location_ContactCenter_Type_K12School:
142  result = make_shared<K12School>(id);
143  typeId = stride::ContactType::Id::K12School;
144  break;
145  case proto::GeoGrid_Location_ContactCenter_Type_PrimaryCommunity:
146  result = make_shared<PrimaryCommunity>(id);
147  typeId = stride::ContactType::Id::PrimaryCommunity;
148  break;
149  case proto::GeoGrid_Location_ContactCenter_Type_SecondaryCommunity:
150  result = make_shared<SecondaryCommunity>(id);
151  typeId = stride::ContactType::Id::SecondaryCommunity;
152  break;
153  case proto::GeoGrid_Location_ContactCenter_Type_College:
154  result = make_shared<College>(id);
155  typeId = stride::ContactType::Id::College;
156  break;
157  case proto::GeoGrid_Location_ContactCenter_Type_Household:
158  result = make_shared<Household>(id);
159  typeId = stride::ContactType::Id::Household;
160  break;
161  case proto::GeoGrid_Location_ContactCenter_Type_Workplace:
162  result = make_shared<Workplace>(id);
163  typeId = stride::ContactType::Id::Workplace;
164  break;
165  break;
166  default: throw runtime_error("No such ContactCenter type");
167  }
168 
169  auto e = make_shared<ThreadException>();
170 #pragma omp parallel
171 #pragma omp single
172  {
173  for (int idx = 0; idx < protoContactCenter.pools_size(); idx++) {
174  const proto::GeoGrid_Location_ContactCenter_ContactPool& protoContactPool =
175  protoContactCenter.pools(idx);
176 #pragma omp task firstprivate(protoContactPool, typeId)
177  {
178  stride::ContactPool* pool = nullptr;
179  e->Run([&protoContactPool, &pool, this, &typeId] {
180  pool = ParseContactPool(protoContactPool, typeId);
181  });
182  if (!e->HasError()) {
183 #pragma omp critical
184  result->RegisterPool(pool);
185  }
186  }
187  }
188 #pragma omp taskwait
189  }
190  e->Rethrow();
191 
192  return result;
193 }

195 stride::ContactPool* GeoGridProtoReader::ParseContactPool(
196  const proto::GeoGrid_Location_ContactCenter_ContactPool& protoContactPool, stride::ContactType::Id type)
197 {
198  // Don't use the id of the ContactPool but the let the Population create an id
199  stride::ContactPool* result;
200 
201 #pragma omp critical
202  result = m_geoGrid->CreateContactPool(type);
203 
204  for (int idx = 0; idx < protoContactPool.people_size(); idx++) {
205  const auto person_id = static_cast<unsigned int>(protoContactPool.people(idx));
206  const auto person = m_people.at(person_id);
207 
208 #pragma omp critical
209  {
210  result->AddMember(person);
211  // Update original pool id with new pool id used in the population
212  person->SetPoolId(type, static_cast<unsigned int>(result->GetId()));
213  }
214  }
215 
216  return result;
217 }

35 void GeoGridProtoWriter::Write(shared_ptr<geopop::GeoGrid> geoGrid, ostream& stream)
36 {
37  GOOGLE_PROTOBUF_VERIFY_VERSION;
38 
39  proto::GeoGrid protoGrid;
40  for (const auto& location : *geoGrid) {
41  WriteLocation(location, protoGrid.add_locations());
42  }
43  for (const auto& person : m_persons_found) {
44  WritePerson(person, protoGrid.add_persons());
45  }
46 
47  m_persons_found.clear();
48  if (!protoGrid.SerializeToOstream(&stream)) {
49  throw stride::util::Exception("There was an error writing the GeoGrid to the file.");
50  }
51  google::protobuf::ShutdownProtobufLibrary();
52  stream.flush();
53 }

90 void GeoGridProtoWriter::WriteLocation(shared_ptr<Location> location, proto::GeoGrid_Location* protoLocation)
91 {
92  protoLocation->set_id(location->GetID());
93  protoLocation->set_name(location->GetName());
94  protoLocation->set_province(location->GetProvince());
95  protoLocation->set_population(location->GetPopCount());
96  auto coordinate = new proto::GeoGrid_Location_Coordinate();
97  WriteCoordinate(location->GetCoordinate(), coordinate);
98  protoLocation->set_allocated_coordinate(coordinate);
99 
100  auto commutes = location->CRefOutgoingCommutes();
101  for (auto commute_pair : commutes) {
102  auto commute = protoLocation->add_commutes();
103  commute->set_to(commute_pair.first->GetID());
104  commute->set_proportion(commute_pair.second);
105  }
106 
107  for (Id typ : IdList) {
108  for (const auto& c : location->RefCenters(typ)) {
109  WriteContactCenter(c, protoLocation->add_contactcenters());
110  }

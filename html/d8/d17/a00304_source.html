44  void Random(T& pools, std::vector<double>& immunityDistribution, double immunityLinkProbability)
45  {
46  // Initialize a vector to count the population per age class [0-100].
47  std::vector<double> populationBrackets(100, 0.0);
48 
49  // Count individuals per age class and set all "susceptible" individuals "immune".
50  // note: focusing on measles, we expect the number of susceptible individuals
51  // to be less compared to the number of immune.
52  // TODO but this is a generic simulator
53  for (auto& c : pools) {
54  for (const auto& p : c.GetPool()) {
55  if (p->GetHealth().IsSusceptible()) {
56  p->GetHealth().SetImmune();
57  populationBrackets[p->GetAge()]++;
58  }
59  }
60  }
61 
62  // Sampler for int in [0, pools.size()) and for double in [0.0, 1.0).
63  const auto poolsSize = static_cast<int>(pools.size());
64  auto intGenerator = m_rn_manager[0].variate_generator(trng::uniform_int_dist(0, poolsSize));
65  auto uniform01Generator = m_rn_manager[0].variate_generator(trng::uniform01_dist<double>());
66 
67  // Calculate the number of susceptible individuals per age class.
68  unsigned int numSusceptible = 0;
69  for (unsigned int age = 0; age < 100; age++) {
70  populationBrackets[age] = floor(populationBrackets[age] * (1 - immunityDistribution[age]));
71  numSusceptible += static_cast<unsigned int>(populationBrackets[age]);
72  }
73 
74  // Sample susceptible individuals, until all age-dependent quota are reached.
75  while (numSusceptible > 0) {
76  // random pool, random order of members
77  const ContactPool& p_pool = pools[intGenerator()];
78  const auto size = static_cast<unsigned int>(p_pool.GetSize());
79  std::vector<unsigned int> indices(size);
80  iota(indices.begin(), indices.end(), 0U);
81  m_rn_manager[0].shuffle(indices.begin(), indices.end());
82 
83  // loop over members, in random order
84  for (unsigned int i_p = 0; i_p < size && numSusceptible > 0; i_p++) {
85  Person& p = *p_pool.GetMember(indices[i_p]);
86  // if p is immune and his/her age class has not reached the quota => make susceptible
87  if (p.GetHealth().IsImmune() && populationBrackets[p.GetAge()] > 0) {
88  p.GetHealth().SetSusceptible();
89  populationBrackets[p.GetAge()]--;
90  numSusceptible--;
91  }
92  // random draw to continue in this pool or to sample a new one
93  if (uniform01Generator() < (1 - immunityLinkProbability)) {
94  break;
95  }
96  }
97  }
98  }

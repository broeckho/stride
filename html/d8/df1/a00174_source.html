30 void TransmissionProfile::Initialize(const ptree& configPt, const ptree& diseasePt)
31 {
32  // Use a quadratic model, fitted to simulation data:
33  // Expected(R0) = (0 + b1*transm_rate + b2*transm_rate^2).
34  const auto r0 = configPt.get<double>("run.r0");
35  const auto b0 = diseasePt.get<double>("disease.transmission.b0");
36  const auto b1 = diseasePt.get<double>("disease.transmission.b1");
37  const auto b2 = diseasePt.get<double>("disease.transmission.b2");
38 
39  // Find root
40  const auto a = b2;
41  const auto b = b1;
42  const auto c = b0 - r0;
43 
44  // To obtain a real values (instead of complex)
45  if (r0 < (-(b * b) / (4 * a))) {
46  const double determ = (b * b) - 4 * a * c;
47  m_transmission_rate = (-b + sqrt(determ)) / (2 * a);
48  } else {
49  throw runtime_error("TransmissionProfile::Initialize> Illegal input values.");
50  }
51 }

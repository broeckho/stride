36 void SecondaryCommunityGenerator::Apply(GeoGrid& geoGrid, const GeoGridConfig& geoGridConfig, unsigned int& ccCounter)
37 {
38  // 1. calculate number of communities, each community has average 2000 persons
39  // 2. assign communities to a location using a discrete distribution reflecting the relative number of
40  // people at that location
41 
42  const auto popCount = geoGridConfig.input.pop_size;
43  const auto communitySize = geoGridConfig.pools.secondary_community_size;
44  const auto communityCount = static_cast<unsigned int>(ceil(popCount / static_cast<double>(communitySize)));
45 
46  vector<double> weights;
47  for (const auto& loc : geoGrid) {
48  const auto weight = static_cast<double>(loc->GetPopCount()) / static_cast<double>(popCount);
49  AssertThrow(weight >= 0 && weight <= 1 && !std::isnan(weight),
50  "SecondaryCommunityGenerator> Invalid weight: " + to_string(weight), m_logger);
51  weights.push_back(weight);
52  }
53 
54  if (weights.empty()) {
55  // trng can't handle empty vectors
56  return;
57  }
58 
59  const auto dist = m_rn_man.GetDiscreteGenerator(weights, 0U);
60  auto& poolSys = geoGrid.GetPopulation()->RefPoolSys();
61 
62  for (auto i = 0U; i < communityCount; i++) {
63  const auto loc = geoGrid[dist()];
64  const auto c = make_shared<ContactCenter>(ccCounter++, Id::SecondaryCommunity);
65 
66  // TODO CheckThisAlgorithm
67  // for (std::size_t j = 0; j < geoGridConfig.pools.pools_per_community; ++j) {
68  if (c->size() == 0) {
69  const auto p = poolSys.CreateContactPool(stride::ContactType::Id::SecondaryCommunity);
70  c->RegisterPool(p);
71  loc->RegisterPool<Id::SecondaryCommunity>(p);
72  }
73 
74  loc->AddCenter(c);
75  }
76 }

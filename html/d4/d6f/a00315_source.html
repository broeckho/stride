48 void StanController::Control()
49 {
50  // -----------------------------------------------------------------------------------------
51  // Prelims.
52  // -----------------------------------------------------------------------------------------
53  CheckEnv();
54  CheckOutputPrefix();
55  InstallLogger();
56  LogStartup();
57 
58  // -----------------------------------------------------------------------------------------
59  // Stan scenario: step 2, make seeds for the stochastic analysis.
60  // -----------------------------------------------------------------------------------------
61  const auto stanCount = m_config.get<unsigned int>("run.stan_count");
62  random_device rd;
63  vector<unsigned int> seeds;
64  for (unsigned int i = 0; i < stanCount; i++) {
65  seeds.emplace_back(rd());
66  }
67  vector<vector<unsigned int>> results(seeds.size());
68 
69  // -----------------------------------------------------------------------------------------
70  // Instantiate simRunners & run, once for each seed.
71  // Multiple runs in parallel, individual runs in single thread.
72  // -----------------------------------------------------------------------------------------
73  m_config.put("run.num_threads", 1);
74 
75 #pragma omp parallel for num_threads(ConfigInfo::NumberAvailableThreads())
76  for (unsigned int i = 0U; i < seeds.size(); ++i) {
77 
78  // ---------------------------------------------------------------------------------
79  // Stan scenario: step 2, build a single stream random number manager for each run.
80  // ---------------------------------------------------------------------------------
81  ptree configPt(m_config);
82  configPt.put("run.rng_seed", seeds[i]);
83  RnMan rnMan{RnInfo{configPt.get<string>("run.rng_seed"), "", 1U}};
84 
85  m_stride_logger->info("Starting run using seed {}", configPt.get<string>("run.rng_seed"));
86 
87  // ---------------------------------------------------------------------------------
88  // Stan scenario: step 3, build a population as specified in config.
89  // ---------------------------------------------------------------------------------
90  auto pop = Population::Create(configPt, rnMan);
91 
92  // -----------------------------------------------------------------------------------------
93  // Sim scenario: step 4, create a simulator, as described by the parameter in the config.
94  // -----------------------------------------------------------------------------------------
95  auto sim = Sim::Create(configPt, pop, rnMan);
96 
97  // ---------------------------------------------------------------------------------
98  // Stan scenario: step 5, build a simulator, register viewer, run, collect results.
99  // ---------------------------------------------------------------------------------
100  auto runner = make_shared<SimRunner>(configPt, sim);
101  auto iViewer = make_shared<viewers::InfectedViewer>(runner);
102  runner->Register(iViewer, bind(&viewers::InfectedViewer::Update, iViewer, std::placeholders::_1));
103  runner->Run();
104  results[i] = iViewer->GetInfectionCounts();
105  }
106 
107  // -----------------------------------------------------------------------------------------
108  // Stan scenario: step 6, output to file.
109  // -----------------------------------------------------------------------------------------
110  const auto numDays = m_config.get<unsigned int>("run.num_days");
111  CSV csv(seeds.begin(), seeds.end());
112  for (unsigned int i = 0U; i < numDays + 1; ++i) {
113  vector<unsigned int> v;
114  for (const auto& res : results) {
115  v.emplace_back(res[i]);
116  }
117  csv.AddRow(v.begin(), v.end());
118  }
119  csv.Write(FileSys::BuildPath(m_config.get<string>("run.output_prefix"), "stan_infected.csv"));
120 
121  // -----------------------------------------------------------------------------------------
122  // Done, shutdown.
123  // -----------------------------------------------------------------------------------------
124  LogShutdown();
125  spdlog::drop_all();
126 }

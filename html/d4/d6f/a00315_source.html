47 void StanController::Control()
48 {
49  // -----------------------------------------------------------------------------------------
50  // Prelims.
51  // -----------------------------------------------------------------------------------------
52  CheckEnv();
53  CheckOutputPrefix();
54  InstallLogger();
55  LogStartup();
56 
57  // -----------------------------------------------------------------------------------------
58  // Seeds for the stochastic analysis.
59  // -----------------------------------------------------------------------------------------
60  const auto stanCount = m_config_pt.get<size_t>("run.stan_count");
61  random_device rd;
62  vector<std::random_device::result_type> seeds;
63  for (unsigned int i = 0; i < stanCount; i++) {
64  seeds.emplace_back(rd());
65  }
66  vector<vector<unsigned int>> results(seeds.size());
67 
68  // -----------------------------------------------------------------------------------------
69  // Instantiate simRunners & run, once for each seed.
70  // Multiple runs in parallel, individual runs not.
71  // -----------------------------------------------------------------------------------------
72  m_config_pt.put("run.num_threads", 1);
73 #pragma omp parallel for num_threads(ConfigInfo::NumberAvailableThreads())
74  for (unsigned int i = 0U; i < seeds.size(); ++i) {
75  ptree configPt(m_config_pt);
76  configPt.put("run.rng_seed", seeds[i]);
77  m_stride_logger->info("Starting run using seed {}", seeds[i]);
78 
79  auto runner =
80  make_shared<SimRunner>(configPt, Population::Create(configPt, m_rn_manager), m_rn_manager);
81  auto iViewer = make_shared<viewers::InfectedViewer>(runner);
82  runner->Register(iViewer, bind(&viewers::InfectedViewer::Update, iViewer, std::placeholders::_1));
83 
84  runner->Run();
85  results[i] = iViewer->GetInfectionCounts();
86  }
87 
88  // -----------------------------------------------------------------------------------------
89  // Output to file.
90  // -----------------------------------------------------------------------------------------
91  const auto numDays = m_config_pt.get<unsigned int>("run.num_days");
92  CSV csv(seeds.begin(), seeds.end());
93  for (unsigned int i = 0U; i < numDays + 1; ++i) {
94  vector<unsigned int> v;
95  for (const auto& res : results) {
96  v.emplace_back(res[i]);
97  }
98  csv.AddRow(v.begin(), v.end());
99  }
100  csv.Write(FileSys::BuildPath(m_config_pt.get<string>("run.output_prefix"), "stan_infected.csv"));
101 
102  // -----------------------------------------------------------------------------------------
103  // Shutdown.
104  // -----------------------------------------------------------------------------------------
105  LogShutdown();
106  spdlog::drop_all();
107 }

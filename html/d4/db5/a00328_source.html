40 shared_ptr<Population> SurveySeeder::Seed(shared_ptr<Population> pop)
41 {
42  const string logLevel = m_config.get<string>("run.contact_log_level", "None");
43  if (logLevel != "None") {
44  Population& population = *pop;
45  auto& poolSys = population.GetContactPoolSys();
46  auto& logger = population.GetContactLogger();
47  const auto popCount = static_cast<unsigned int>(population.size() - 1);
48  const auto numSurveyed = m_config.get<unsigned int>("run.num_participants_survey");
49 
50  assert( (popCount >= 1U) && "SurveySeeder> Population count zero unacceptable.");
51  assert( (popCount >= numSurveyed) && "SurveySeeder> Pop count has to exceeed number of surveyed.");
52 
53  // Use while-loop to get 'participants' unique participants (default sampling is with replacement).
54  // A for loop will not do because we might draw the same person twice.
55  auto numSamples = 0U;
56  auto generator = m_rn_man[0].variate_generator(trng::uniform_int_dist(0, static_cast<int>(popCount)));
57 
58  while (numSamples < numSurveyed) {
59  Person& p = population[generator()];
60  if (p.IsSurveyParticipant()) {
61  continue;
62  }
63  p.ParticipateInSurvey();
64 
65  const auto h = p.GetHealth();
66  logger->info(
67  "[PART] {} {} {} {} {} {} {} {} {} {} {} {} {} {} {} {} {} {} {} {} {}",
68  p.GetId(), p.GetAge(), p.GetPoolId(Id::Household),
69  p.GetPoolId(Id::K12School), p.GetPoolId(Id::College),
70  p.GetPoolId(Id::Workplace), h.IsSusceptible(), h.IsInfected(), h.IsInfectious(),
71  h.IsRecovered(), h.IsImmune(), h.GetStartInfectiousness(), h.GetStartSymptomatic(),
72  h.GetEndInfectiousness(), h.GetEndSymptomatic(),
73  poolSys[Id::Household][p.GetPoolId(Id::Household)].GetSize(),
74  poolSys[Id::K12School][p.GetPoolId(Id::K12School)].GetSize(),
75  poolSys[Id::College][p.GetPoolId(Id::College)].GetSize(),
76  poolSys[Id::Workplace][p.GetPoolId(Id::Workplace)].GetSize(),
77  poolSys[Id::PrimaryCommunity][p.GetPoolId(Id::PrimaryCommunity)].GetSize(),
78  poolSys[Id::SecondaryCommunity][p.GetPoolId(Id::SecondaryCommunity)].GetSize());
79 
80  numSamples++;
81  }
82  }
83  return pop;
84 }

96 void ControlHelper::LogStartup()
97 {
98  m_stride_logger->info("{} starting up at: {}", m_name, TimeStamp().ToString());
99  m_stride_logger->info("Executing revision {}", ConfigInfo::GitRevision());
100  m_stride_logger->info("Creating dir: {}", m_output_prefix);
101  m_stride_logger->trace("Executing: {}", FileSys::GetExecPath().string());
102  m_stride_logger->trace("Current directory: {}", FileSys::GetCurrentDir().string());
103  if (m_use_install_dirs) {
104  m_stride_logger->trace("Install directory: {}", FileSys::GetRootDir().string());
105  m_stride_logger->trace("Config directory: {}", FileSys::GetConfigDir().string());
106  m_stride_logger->trace("Data directory: {}", FileSys::GetDataDir().string());
107  }
108  if (ConfigInfo::HaveOpenMP()) {
109  m_stride_logger->info("Max number OpenMP threads in this environment: {}",
110  ConfigInfo::NumberAvailableThreads());
111  m_stride_logger->info("Configured number of threads: {}",
112  m_config_pt.get<unsigned int>("run.num_threads"));
113  } else {
114  m_stride_logger->info("Not using OpenMP threads.");
115  }
116 }

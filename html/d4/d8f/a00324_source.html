40 void WorkplacePopulator::Apply(shared_ptr<GeoGrid> geoGrid, GeoGridConfig& geoGridConfig)
41 {
42  m_logger->info("Starting to populate Workplaces");
43 
44  m_geoGrid = geoGrid;
45  m_geoGridConfig = geoGridConfig;
46  m_fractionCommutingStudents = 0;
47  m_workplacesInCity.clear();
48  m_currentLoc = nullptr;
49  m_assignedTo0 = 0;
50  m_assignedCommuting = 0;
51  m_assignedNotCommuting = 0;
52  m_distNonCommuting = discreteDist();
53  m_nearByWorkplaces.clear();
54  m_disCommuting = discreteDist();
55  m_commutingLocations.clear();
56 
57  CalculateFractionCommutingStudents();
58  CalculateWorkplacesInCity();
59 
60  // for every location
61  for (const auto& loc : *geoGrid) {
62  if (loc->GetPopulation() == 0) {
63  continue;
64  }
65  m_currentLoc = loc;
66  CalculateCommutingLocations();
67  CalculateNearbyWorkspaces();
68 
69  // 2. for every worker assign a class
70  for (const auto& household : loc->GetContactCentersOfType<Household>()) {
71  auto contactPool = household->GetPools()[0];
72  for (auto p : *contactPool) {
73  if (p->IsWorkableCandidate()) {
74  bool isStudent =
75  MakeChoice(geoGridConfig.input.fraction_1826_years_WhichAreStudents);
76  bool isActiveWorker =
77  MakeChoice(geoGridConfig.input.fraction_1865_years_active);
78 
79  if ((p->IsCollegeStudentCandidate() && !isStudent) || isActiveWorker) {
80  AssignActive(p);
81  } else {
82  // this person isn't an active employee
83  p->SetWorkId(0);
84  m_assignedTo0++;
85  }
86  }
87  }
88  }
89  }
90 
91  m_logger->info("Populated workplaces, assigned to 0 {}, assigned (commuting) {} assigned (not commuting) {} ",
92  m_assignedTo0, m_assignedCommuting, m_assignedNotCommuting);
93 }

95 void WorkplacePopulator::CalculateFractionCommutingStudents()
96 {
97  m_fractionCommutingStudents = 0;
98  if (static_cast<bool>(m_geoGridConfig.input.fraction_active_commutingPeople) &&
99  m_geoGridConfig.calculated.popcount_1865_and_years_active) {
100  m_fractionCommutingStudents = (m_geoGridConfig.calculated.popcount_1826_years_and_student *
101  m_geoGridConfig.input.fraction_student_commutingPeople) /
102  (m_geoGridConfig.calculated.popcount_1865_and_years_active *
103  m_geoGridConfig.input.fraction_active_commutingPeople);
104  }
105 }

107 void WorkplacePopulator::CalculateWorkplacesInCity()
108 {
109  for (const shared_ptr<Location>& loc : *m_geoGrid) {
110  vector<ContactPool*> contactPools;
111  for (const auto& wp : loc->GetContactCentersOfType<Workplace>()) {
112  contactPools.insert(contactPools.end(), wp->begin(), wp->end());
113  }
114 
115  auto disPools = m_rnManager[0].variate_generator(
116  trng::uniform_int_dist(0, static_cast<trng::uniform_int_dist::result_type>(contactPools.size())));
117 
118  m_workplacesInCity[loc.get()] = {contactPools, disPools};
119  }
120 }

122 void WorkplacePopulator::AssignActive(Person* person)
123 {
124  // this person is (student and active) or active
125  if (!m_commutingLocations.empty() && MakeChoice(m_geoGridConfig.input.fraction_active_commutingPeople)) {
126  // this person is commuting
127  const auto& info = m_workplacesInCity[m_commutingLocations[m_disCommuting()]];
128  const auto id = info.second(); // id of the location this person is commuting to
129  info.first[id]->AddMember(person);
130  person->SetWorkId(info.first[id]->GetId());
131  m_assignedCommuting++;
132  } else {
133  // this person is not commuting
134  const auto id = m_distNonCommuting();
135  m_nearByWorkplaces[id]->AddMember(person);
136  person->SetWorkId(static_cast<unsigned int>(m_nearByWorkplaces[id]->GetId()));
137  m_assignedNotCommuting++;
138  }
139 }

141 void WorkplacePopulator::CalculateCommutingLocations()
142 {
143  // find all Workplaces were employees from this location commute to
144  m_commutingLocations.clear();
145  m_disCommuting = discreteDist();
146 
147  vector<double> commutingWeights;
148  for (const pair<Location*, double>& commute : m_currentLoc->GetOutgoingCommuningCities()) {
149  const auto& Workplaces = commute.first->GetContactCentersOfType<Workplace>();
150  if (!Workplaces.empty()) {
151  m_commutingLocations.push_back(commute.first);
152  const auto weight = commute.second - (commute.second * m_fractionCommutingStudents);
153  commutingWeights.push_back(weight);
154  ExcAssert(weight >= 0 && weight <= 1 && !isnan(weight),
155  "Invalid weight due to data in WorkplacePopulator, weight: " + to_string(weight));
156  }
157  }
158 
159  if (!commutingWeights.empty()) {
160  m_disCommuting = m_rnManager[0].variate_generator(
161  trng::discrete_dist(commutingWeights.begin(), commutingWeights.end()));
162  }
163 }

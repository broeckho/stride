46 void GeoGridJSONReader::Read()
47 {
48  boost::property_tree::ptree root;
49  try {
50  boost::property_tree::read_json(*m_inputStream, root);
51  } catch (runtime_error&) {
52  throw Exception("Problem parsing JSON file, check whether empty or invalid JSON.");
53  }
54 
55  auto& geoGrid = m_population->RefGeoGrid();
56  auto people = root.get_child("persons");
57 
58 #pragma omp parallel
59 #pragma omp single
60  {
61  for (auto it = people.begin(); it != people.end(); it++) {
62 #pragma omp task firstprivate(it)
63  {
64 #pragma omp critical
65  {
66  auto person = ParsePerson(it->second.get_child(""));
67  m_people[person->GetId()] = person;
68  }
69  }
70  }
71 #pragma omp taskwait
72  }
73  auto locations = root.get_child("locations");
74  auto e = make_shared<ThreadException>();
75 #pragma omp parallel
76 #pragma omp single
77  {
78  for (auto it = locations.begin(); it != locations.end(); it++) {
79  shared_ptr<Location> loc;
80 #pragma omp task firstprivate(it, loc)
81  {
82  e->Run([&loc, this, &it] { loc = ParseLocation(it->second.get_child("")); });
83  if (!e->HasError())
84 #pragma omp critical
85  geoGrid.AddLocation(move(loc));
86  }
87  }
88 #pragma omp taskwait
89  }
90  e->Rethrow();
91  AddCommutes(geoGrid);
92  m_commutes.clear();
93  m_people.clear();
94 }

96 shared_ptr<Location> GeoGridJSONReader::ParseLocation(boost::property_tree::ptree& location)
97 {
98  const auto id = boost::lexical_cast<unsigned int>(location.get<string>("id"));
99  const auto name = location.get<string>("name");
100  const auto province = boost::lexical_cast<unsigned int>(location.get<string>("province"));
101  const auto population = boost::lexical_cast<unsigned int>(location.get<string>("population"));
102  const auto coordinate = ParseCoordinate(location.get_child("coordinate"));
103 
104  auto result = make_shared<Location>(id, province, coordinate, name, population);
105  auto contactCenters = location.get_child("contactCenters");
106  auto e = make_shared<ThreadException>();
107 
108 #pragma omp parallel
109 #pragma omp single
110  {
111  for (auto it = contactCenters.begin(); it != contactCenters.end(); it++) {
112  shared_ptr<ContactCenter> center;
113 #pragma omp task firstprivate(it, center)
114  {
115  e->Run([&it, this, &center] { center = ParseContactCenter(it->second.get_child("")); });
116  if (!e->HasError())
117 #pragma omp critical
118  result->AddCenter(center);
119  }
120  }
121 #pragma omp taskwait
122  }
123  e->Rethrow();
124 
125  if (location.count("commutes")) {
126  boost::property_tree::ptree commutes = location.get_child("commutes");
127  for (auto it = commutes.begin(); it != commutes.end(); it++) {
128  const auto to = boost::lexical_cast<unsigned int>(it->first);
129  const auto amount = boost::lexical_cast<double>(it->second.data());
130 #pragma omp critical
131  m_commutes.emplace_back(id, to, amount);
132  }
133  }
134 
135  return result;
136 }

145 shared_ptr<ContactCenter> GeoGridJSONReader::ParseContactCenter(boost::property_tree::ptree& contactCenter)
146 {
147  const auto type = contactCenter.get<string>("type");
148  const auto id = boost::lexical_cast<unsigned int>(contactCenter.get<string>("id"));
149 
150  shared_ptr<ContactCenter> result;
151  ContactType::Id typeId;
152  if (type == ToString(Id::K12School)) {
153  result = make_shared<K12SchoolCenter>(id);
154  typeId = Id::K12School;
155  } else if (type == ToString(Id::College)) {
156  result = make_shared<CollegeCenter>(id);
157  typeId = Id::College;
158  } else if (type == ToString(Id::Household)) {
159  result = make_shared<HouseholdCenter>(id);
160  typeId = Id::Household;
161  } else if (type == ToString(Id::PrimaryCommunity)) {
162  result = make_shared<PrimaryCommunityCenter>(id);
163  typeId = Id::PrimaryCommunity;
164  } else if (type == ToString(Id::SecondaryCommunity)) {
165  result = make_shared<SecondaryCommunityCenter>(id);
166  typeId = Id::SecondaryCommunity;
167  } else if (type == ToString(Id::Workplace)) {
168  result = make_shared<WorkplaceCenter>(id);
169  typeId = Id::Workplace;
170  } else {
171  throw Exception("No such ContactCenter type: " + type);
172  }
173 
174  auto contactPools = contactCenter.get_child("pools");
175  auto e = make_shared<ThreadException>();
176 
177 #pragma omp parallel
178 #pragma omp single
179  {
180  for (auto it = contactPools.begin(); it != contactPools.end(); it++) {
181 #pragma omp task firstprivate(it)
182  {
183  ContactPool* pool = nullptr;
184  e->Run([&it, &pool, this, typeId] {
185  pool = ParseContactPool(it->second.get_child(""), typeId);
186  });
187  if (!e->HasError()) {
188 #pragma omp critical
189  result->RegisterPool(pool);
190  }
191  }
192  }
193 #pragma omp taskwait
194  }
195  e->Rethrow();
196  return result;
197 }

199 ContactPool* GeoGridJSONReader::ParseContactPool(boost::property_tree::ptree& contactPool, ContactType::Id typeId)
200 {
201  // Don't use the id of the ContactPool but the let the Population create an id.
202  auto result = m_population->RefPoolSys().CreateContactPool(typeId);
203  auto people = contactPool.get_child("people");
204 
205  for (auto it = people.begin(); it != people.end(); it++) {
206  auto person_id = boost::lexical_cast<unsigned int>(it->second.get<string>(""));
207  if (m_people.count(person_id) == 0) {
208  throw Exception("No such person: " + to_string(person_id));
209  }
210 #pragma omp critical
211  result->AddMember(m_people[person_id]);
212  }
213 
214  return result;
215 }

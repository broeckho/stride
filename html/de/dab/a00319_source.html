47 void StanController::Control()
48 {
49  // -----------------------------------------------------------------------------------------
50  // Prelims.
51  // -----------------------------------------------------------------------------------------
52  CheckEnv();
53  CheckOutputPrefix();
54  InstallLogger();
55  LogStartup();
56 
57  // -----------------------------------------------------------------------------------------
58  // Seeds for the stochastic analysis.
59  // -----------------------------------------------------------------------------------------
60  const auto stanCount = m_config.get<unsigned int>("run.stan_count");
61  random_device rd;
62  vector<std::random_device::result_type> seeds;
63  for (unsigned int i = 0; i < stanCount; i++) {
64  seeds.emplace_back(rd());
65  }
66  vector<vector<unsigned int>> results(seeds.size());
67 
68  // -----------------------------------------------------------------------------------------
69  // Instantiate simRunners & run, once for each seed.
70  // Multiple runs in parallel, individual runs in single thread..
71  // -----------------------------------------------------------------------------------------
72  m_config.put("run.num_threads", 1);
73 
74 #pragma omp parallel for num_threads(ConfigInfo::NumberAvailableThreads())
75  for (unsigned int i = 0U; i < seeds.size(); ++i) {
76  ptree configPt(m_config);
77  configPt.put("run.rng_seed", seeds[i]);
78  m_stride_logger->info("Starting run using seed {}", seeds[i]);
79 
80  auto pop = Population::Create(configPt, m_rn_man);
81  auto runner = make_shared<SimRunner>(configPt, pop, m_rn_man);
82 
83  auto iViewer = make_shared<viewers::InfectedViewer>(runner);
84  runner->Register(iViewer, bind(&viewers::InfectedViewer::Update, iViewer, std::placeholders::_1));
85 
86  runner->Run();
87  results[i] = iViewer->GetInfectionCounts();
88  }
89 
90  // -----------------------------------------------------------------------------------------
91  // Output to file.
92  // -----------------------------------------------------------------------------------------
93  const auto numDays = m_config.get<unsigned int>("run.num_days");
94  CSV csv(seeds.begin(), seeds.end());
95  for (unsigned int i = 0U; i < numDays + 1; ++i) {
96  vector<unsigned int> v;
97  for (const auto& res : results) {
98  v.emplace_back(res[i]);
99  }
100  csv.AddRow(v.begin(), v.end());
101  }
102  csv.Write(FileSys::BuildPath(m_config.get<string>("run.output_prefix"), "stan_infected.csv"));
103 
104  // -----------------------------------------------------------------------------------------
105  // Shutdown.
106  // -----------------------------------------------------------------------------------------
107  LogShutdown();
108  spdlog::drop_all();
109 }

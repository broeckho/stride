31 void CommunityGenerator::Apply(shared_ptr<GeoGrid> geoGrid, const GeoGridConfig& geoGridConfig,
32  unsigned int& contactCenterCounter)
33 {
34  // 1. calculate number of communities, each community has average 2000 persons
35  // 2. assign communities to a location using a discrete distribution reflecting the relative number of
36  // people at that location
37 
38  const auto popCount = geoGridConfig.input.pop_size;
39  const auto communityCount =
40  static_cast<unsigned int>(ceil(popCount / static_cast<double>(geoGridConfig.pools.community_size)));
41 
42  vector<double> weights;
43  for (const auto& loc : *geoGrid) {
44  const auto weight = static_cast<double>(loc->GetPopCount()) / static_cast<double>(popCount);
45  CheckWeight("CommunityGenerator", weight);
46  weights.push_back(weight);
47  }
48 
49  if (weights.empty()) {
50  // trng can't handle empty vectors
51  return;
52  }
53 
54  const auto dist = m_rnManager[0].variate_generator(trng::discrete_dist(weights.begin(), weights.end()));
55 
56  for (auto i = 0U; i < communityCount; i++) {
57  const auto loc = (*geoGrid)[dist()];
58  const auto pc = make_shared<PrimaryCommunity>(contactCenterCounter++);
59  pc->Fill(geoGridConfig, geoGrid);
60  loc->AddContactCenter(pc);
61  }
62  for (auto i = 0U; i < communityCount; i++) {
63  const auto loc = (*geoGrid)[dist()];
64  const auto sc = make_shared<SecondaryCommunity>(contactCenterCounter++);
65  sc->Fill(geoGridConfig, geoGrid);
66  loc->AddContactCenter(sc);
67  }
68 }

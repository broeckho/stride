37 void CollegePopulator::Apply(shared_ptr<GeoGrid> geoGrid, const GeoGridConfig& geoGridConfig)
38 {
39  m_logger->info("Starting to populate Colleges");
40 
41  set<ContactPool*> found;
42  auto students = 0U;
43  auto commuting = 0U;
44 
45  // for every location
46  for (const auto& loc : *geoGrid) {
47  if (loc->GetPopCount() == 0) {
48  continue;
49  }
50  // 1. find all highschools in an area of 10-k*10 km
51  const auto& nearByColleges = GetNearbyPools<College>(geoGrid, loc);
52 
53  AssertThrow(!nearByColleges.empty(), "No HighSchool found!", m_logger);
54 
55  const auto distNonCommuting = m_rnManager[0].variate_generator(
56  trng::uniform_int_dist(0, static_cast<trng::uniform_int_dist::result_type>(nearByColleges.size())));
57 
58  // 2. find all colleges where students from this location commute to
59  vector<Location*> commutingCollege;
60  vector<double> commutingWeights;
61  for (const auto& commute : loc->GetOutgoingCommutingCities()) {
62  const auto& highSchools = commute.first->GetContactCentersOfType<College>();
63  if (!highSchools.empty()) {
64  commutingCollege.push_back(commute.first);
65  commutingWeights.push_back(commute.second);
66  }
67  }
68 
69  function<trng::discrete_dist::result_type()> disCommuting;
70 
71  if (!commutingWeights.empty()) {
72  disCommuting = m_rnManager[0].variate_generator(
73  trng::discrete_dist(commutingWeights.begin(), commutingWeights.end()));
74  }
75 
76  // 2. for every student assign a class
77  for (const auto& household : loc->GetContactCentersOfType<Household>()) {
78  ContactPool* contactPool = household->GetPools()[0];
79  found.insert(contactPool);
80  for (Person* p : *contactPool) {
81  if (AgeBrackets::College::HasAge(p->GetAge()) &&
82  MakeChoice(geoGridConfig.input.participation_college)) {
83  students++;
84  // this person is a student
85  if (!commutingCollege.empty() &&
86  MakeChoice(geoGridConfig.input.fraction_college_commuters)) {
87  // this person is commuting
88  commuting++;
89 
90  // id of the location this person is commuting to
91  auto locationId = disCommuting();
92  // create list of classes for each highschool at this location
93  const auto& highSchools =
94  commutingCollege[locationId]->GetContactCentersOfType<College>();
95 
96  vector<ContactPool*> contactPools;
97  for (const auto& hs : highSchools) {
98  contactPools.insert(contactPools.end(), hs->begin(), hs->end());
99  }
100 
101  auto disPools = m_rnManager[0].variate_generator(trng::uniform_int_dist(
102  0, static_cast<trng::uniform_int_dist::result_type>(
103  contactPools.size())));
104 
105  auto id = disPools();
106  contactPools[id]->AddMember(p);
107  p->SetPoolId(Id::College, contactPools[id]->GetId());
108  } else {
109  auto id = distNonCommuting();
110  nearByColleges[id]->AddMember(p);
111  p->SetPoolId(Id::College, nearByColleges[id]->GetId());
112  }
113  }
114  }
115  }
116  }
117  m_logger->info("Number of students in Colleges: {}", students);
118  m_logger->info("Number of classes: {}", found.size());
119  m_logger->info("Number students that commute: ", commuting);
120 }

36 void Populator<stride::ContactType::Id::Workplace>::Apply(GeoGrid& geoGrid, const GeoGridConfig& geoGridConfig)
37 {
38  m_logger->trace("Starting to populate Workplaces");
39 
40  auto genCommute{function<int()>()};
41  auto genNonCommute{function<int()>()};
42  vector<ContactPool*> nearbyWp{};
43  vector<Location*> commuteLocations{};
44 
45  const auto participCollege = geoGridConfig.param.participation_college;
46  const auto participWorkplace = geoGridConfig.param.particpation_workplace;
47  const auto popCollege = geoGridConfig.info.popcount_college;
48  const auto popWorkplace = geoGridConfig.info.popcount_workplace;
49  const auto fracCollegeCommute = geoGridConfig.param.fraction_college_commuters;
50  const auto fracWorkplaceCommute = geoGridConfig.param.fraction_workplace_commuters;
51 
52  double fracCommuteStudents = 0.0;
53  if (static_cast<bool>(fracWorkplaceCommute) && popWorkplace) {
54  fracCommuteStudents = (popCollege * fracCollegeCommute) /(popWorkplace * fracWorkplaceCommute);
55  }
56 
57  // --------------------------------------------------------------------------------
58  // For every location, if populated ...
59  // --------------------------------------------------------------------------------
60  for (const auto& loc : geoGrid) {
61  if (loc->GetPopCount() == 0) {
62  continue;
63  }
64 
65  // --------------------------------------------------------------------------------
66  // Find all Workplaces were employees from this location commute to
67  // --------------------------------------------------------------------------------
68  commuteLocations.clear();
69  genCommute = function<int()>();
70 
71  vector<double> commutingWeights;
72  for (const pair<Location*, double>& commute : loc->CRefOutgoingCommutes()) {
73  const auto& workplaces = commute.first->RefPools(Id::Workplace);
74  if (!workplaces.empty()) {
75  commuteLocations.push_back(commute.first);
76  const auto weight = commute.second - (commute.second * fracCommuteStudents);
77  commutingWeights.push_back(weight);
78  AssertThrow(weight >= 0.0 && weight <= 1.0 && !isnan(weight),
79  "Invalid weight: " + to_string(weight), m_logger);
80  }
81  }
82 
83  if (!commutingWeights.empty()) {
84  genCommute = m_rn_man.GetDiscreteGenerator(commutingWeights, 0U);
85  }
86 
87  // --------------------------------------------------------------------------------
88  // Set NearbyWorkspacePools and associated generator
89  // --------------------------------------------------------------------------------
90  nearbyWp = geoGrid.GetNearbyPools(Id::Workplace, *loc);
91  genNonCommute = m_rn_man.GetUniformIntGenerator(0, static_cast<int>(nearbyWp.size()), 0U);
92 
93  // --------------------------------------------------------------------------------
94  // For everyone of working age: decide between work or college (iff of College age)
95  // --------------------------------------------------------------------------------
96  for (auto& hhPool : loc->RefPools(Id::Household)) {
97  for (auto person : *hhPool) {
98  if (!Workplace::HasAge(person->GetAge())) {
99  continue;
100  }
101 
102  bool isStudent = m_rn_man.MakeWeightedCoinFlip(participCollege);
103  bool isActiveWorker = m_rn_man.MakeWeightedCoinFlip(participWorkplace);
104 
105  if ((College::HasAge(person->GetAge()) && !isStudent) || isActiveWorker) {
106  // ---------------------------------------------
107  // this person is employed
108  // ---------------------------------------------
109  const auto isCommuter = m_rn_man.MakeWeightedCoinFlip(fracWorkplaceCommute);
110  if (!commuteLocations.empty() && isCommuter) {
111  // --------------------------------------------------------------
112  // this person commutes to the Location and in particular to Pool
113  // --------------------------------------------------------------
114  auto& pools = commuteLocations[genCommute()]->RefPools(Id::Workplace);
115  auto s = static_cast<int>(pools.size());
116  auto gen = m_rn_man.GetUniformIntGenerator(0, s);
117  auto pool = pools[gen()];
118  // so that's it
119  pool->AddMember(person);
120  person->SetPoolId(Id::Workplace, pool->GetId());
121  } else {
122  // ----------------------------
123  // this person does not commute
124  // ----------------------------
125  const auto idraw = genNonCommute();
126  nearbyWp[idraw]->AddMember(person);
127  person->SetPoolId(Id::Workplace, nearbyWp[idraw]->GetId());
128  }
129  } else {
130  // -----------------------------
131  // this person has no employment
132  // -----------------------------
133  person->SetPoolId(Id::Workplace, 0);
134  }
135  }
136  }
137  }
138 
139  m_logger->trace("Done populating Workplaces");
140 }

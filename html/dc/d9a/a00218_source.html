33 void GeoGridProtoWriter::Write(shared_ptr<geopop::GeoGrid> geoGrid, ostream& stream)
34 {
35  GOOGLE_PROTOBUF_VERIFY_VERSION;
36 
37  proto::GeoGrid protoGrid;
38  for (const auto& location : *geoGrid) {
39  WriteLocation(location, protoGrid.add_locations());
40  }
41  for (const auto& person : m_persons_found) {
42  WritePerson(person, protoGrid.add_persons());
43  }
44 
45  m_persons_found.clear();
46  if (!protoGrid.SerializeToOstream(&stream)) {
47  throw stride::util::Exception("There was an error writing the GeoGrid to the file.");
48  }
49  google::protobuf::ShutdownProtobufLibrary();
50  stream.flush();
51 }

53 void GeoGridProtoWriter::WriteLocation(shared_ptr<Location> location, proto::GeoGrid_Location* protoLocation)
54 {
55  protoLocation->set_id(location->GetID());

57  protoLocation->set_province(location->GetProvince());
58  protoLocation->set_population(location->GetPopCount());

61  protoLocation->set_allocated_coordinate(coordinate);
62 
63  auto commutes = location->GetOutgoingCommutingCities();
64  for (auto commute_pair : commutes) {

66  commute->set_to(commute_pair.first->GetID());

68  }

70  for (const auto& contactCenter : *location) {
71  WriteContactCenter(contactCenter, protoLocation->add_contactcenters());

38 void PtreeViewer<T>::Update(const myhayai::event::Payload& payload)
39 {
40  static unsigned int exec = 0U;
41  static unsigned int disabled = 0U;
42  const string name = payload.m_test_name;
43  static Stopwatch<> clock("total");
44 
45  if (payload.m_id == event::Id::SkipTest) {
46  ++disabled;
47  }
48  if (payload.m_id == event::Id::BeginTest) {
49  ++exec;
50  }
51 
52  switch (payload.m_id) {
53  case event::Id::BeginBench: {
54  clock.Start();
55  m_ptree.put("benchmark.date", TimeStamp().ToTag());
56  m_ptree.put("benchmark.host", ConfigInfo::GetHostname());
57  m_ptree.put("benchmark.git_comit", ConfigInfo::GitRevision());
58  break;
59  }
60  case event::Id::EndBench: {
61  m_ptree.put("benchmark.executed", exec);
62  m_ptree.put("benchmark.disabled", disabled);
63  m_ptree.put("benchmark.excluded", m_descriptors.size() - exec - disabled);
64  const auto t = duration_cast<T>(clock.Stop().Get());
65  m_ptree.put("benchmark.total_time", TimeToString::ToColonString(t));
66  break;
67  }
68  case event::Id::SkipTest: {
69  ptree pt;
70  pt.put("name", name);
71  pt.put("status", "disabled");
72  m_ptree.add_child("benchmark.test", pt);
73  break;
74  }
75  case event::Id::AbortTest: {
76  ptree pt;
77  pt.put("name", name);
78  pt.put("status", "aborted");
79  pt.put("message", payload.m_msg);
80  m_ptree.add_child("benchmark.test", pt);
81  break;
82  }
83  case event::Id::BeginTest: {
84  break;
85  }
86  case event::Id::EndTest: {
87  using valtyp = typename decltype(payload.m_run_times)::value_type;
88  const auto stats = BoxPlotData<valtyp>::Calculate(payload.m_run_times);
89  const auto median = duration_cast<T>(stats.m_median);
90  const auto min = duration_cast<T>(stats.m_min);
91  const auto max = duration_cast<T>(stats.m_max);
92  const auto quart1 = duration_cast<T>(stats.m_quartile1);
93  const auto quart3 = duration_cast<T>(stats.m_quartile3);
94 
95  TestDescriptor t_d = m_descriptors[name];
96  ptree pt;
97  pt.put("name", name);
98  if (t_d.m_info_factory) {
99  pt.add_child("info", t_d.m_info_factory());
100  }
101  pt.put("status", "executed");
102  pt.put("run_count", payload.m_run_times.size());
103  for (auto& rt : payload.m_run_times) {
104  pt.add("run_times.run", duration_cast<T>(rt).count());
105  }
106  pt.put("boxplot.median", median.count());
107  pt.put("boxplot.min", min.count());
108  pt.put("boxplot.max", max.count());
109  pt.put("boxplot.quartile1", quart1.count());
110  pt.put("boxplot.quartile3", quart3.count());
111 
112  m_ptree.add_child("benchmark.test", pt);
113  break;
114  }
115  }

45 std::shared_ptr<Population> Population::Create(const boost::property_tree::ptree& config, util::RnMan rnMan,
46  std::shared_ptr<spdlog::logger> strideLogger)
47 {
48  if (!strideLogger) {
49  strideLogger = LogUtils::CreateNullLogger("Population_logger");
50  }
51 
52  // --------------------------------------------------------------
53  // Create empty population & and give it a ContactLogger.
54  // --------------------------------------------------------------
55  const auto pop = Create();
56  if (config.get<bool>("run.contact_output_file", true)) {
57  const auto prefix = config.get<string>("run.output_prefix");
58  const auto logPath = FileSys::BuildPath(prefix, "contact_log.txt");
59  pop->RefContactLogger() = LogUtils::CreateRotatingLogger("contact_logger", logPath.string());
60  pop->RefContactLogger()->set_pattern("%v");
61  strideLogger->info("Contact logging requested; logger set up.");
62  } else {
63  pop->RefContactLogger() = LogUtils::CreateNullLogger("contact_logger");
64  strideLogger->info("No contact logging requested.");
65  }
66 
67  // -----------------------------------------------------------------------------------------
68  // Build population.
69  // -----------------------------------------------------------------------------------------
70  std::string pop_type = config.get<std::string>("run.population_type", "default");
71  if (pop_type == "import") {
72  strideLogger->info("Invoking ImportPopBuilder.");
73  ImportPopBuilder(config, rnMan, strideLogger).Build(pop);
74  } else if (pop_type == "generate") {
75  strideLogger->info("Invoking GeoPopBuilder.");
76  GeoPopBuilder(config, rnMan, strideLogger).Build(pop);
77  } else {
78  strideLogger->info("Invoking DefaultPopBuilder.");
79  DefaultPopBuilder(config, rnMan, strideLogger).Build(pop);
80  }
81 
82  // -----------------------------------------------------------------------------------------
83  // Done.
84  // -----------------------------------------------------------------------------------------
85  return pop;
86 }

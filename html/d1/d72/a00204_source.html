47 void GeoGridBuilder::GenCities(const GeoGridConfig& geoGridConfig, const string& citiesFileName,
48  const string& commutingFileName)
49 {
50  m_geoGrid = make_shared<GeoGrid>(m_population.get());
51 
52  ReaderFactory readerFactory;
53  shared_ptr<CitiesReader> citiesReader;
54  shared_ptr<CommutesReader> commutesReader;
55 
56  citiesReader = readerFactory.CreateCitiesReader(citiesFileName);
57  citiesReader->FillGeoGrid(m_geoGrid);
58  if (!commutingFileName.empty()) {
59  commutesReader = readerFactory.CreateCommutesReader(commutingFileName);
60  commutesReader->FillGeoGrid(m_geoGrid);
61  }
62 
63  for (const shared_ptr<Location>& loc : *m_geoGrid) {
64  loc->SetPopCount(geoGridConfig.input.pop_size);
65  }
66  m_geoGrid->Finalize();

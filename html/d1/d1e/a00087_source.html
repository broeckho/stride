43 std::tuple<bool, size_t> ContactPool::SortMembers()
44 {
45  bool infectious_cases = false;
46  size_t num_cases = 0;
47 
48  for (size_t i_member = 0; i_member < m_index_immune; i_member++) {
49  // if immune, move to back
50  if (m_members[i_member]->GetHealth().IsImmune()) {
51  bool swapped = false;
52  size_t new_place = m_index_immune - 1;
53  m_index_immune--;
54  while (!swapped && new_place > i_member) {
55  if (m_members[new_place]->GetHealth().IsImmune()) {
56  m_index_immune--;
57  new_place--;
58  } else {
59  swap(m_members[i_member], m_members[new_place]);
60  swapped = true;
61  }
62  }
63  }
64  // else, if not susceptible, move to front
65  else if (!m_members[i_member]->GetHealth().IsSusceptible()) {
66  if (!infectious_cases && m_members[i_member]->GetHealth().IsInfectious()) {
67  infectious_cases = true;
68  }
69  if (i_member > num_cases) {
70  swap(m_members[i_member], m_members[num_cases]);
71  }
72  num_cases++;
73  }
74  }
75  return std::make_tuple(infectious_cases, num_cases);
76 }

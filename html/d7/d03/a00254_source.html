121 inline double GetContactRate(const AgeContactProfile& profile, const Person* p, size_t pool_size)
122 {
123  const double reference_num_contacts{profile[EffectiveAge(static_cast<unsigned int>(p->GetAge()))]};
124  const double potential_num_contacts{static_cast<double>(pool_size - 1)};
125 
126  double individual_contact_rate = reference_num_contacts / potential_num_contacts;
127  if (individual_contact_rate >= 1) {
128  individual_contact_rate = 0.999;
129  }
130  // Contacts are reciprocal, so one needs to apply only half of the contacts here.
131  individual_contact_rate = individual_contact_rate / 2;
132  // Contacts are bi-directional: contact probability for 1=>2 and 2=>1 = indiv_cnt_rate*indiv_cnt_rate
133  individual_contact_rate += (individual_contact_rate * individual_contact_rate);
134 
135  return individual_contact_rate;
136 }

146 template <ContactLogMode::Id LL, bool TIC, bool TO>
147 void Infector<LL, TIC, TO>::Exec(ContactPool& pool, const AgeContactProfile& profile,
148  const TransmissionProfile& transProfile, ContactHandler& cHandler,
149  unsigned short int simDay, shared_ptr<spdlog::logger> cLogger)
150 {
151  using LP = LOG_POLICY<LL>;
152 
153  // set up some stuff
154  const auto pType = pool.m_pool_type;
155  const auto& pMembers = pool.m_members;
156  const auto pSize = pMembers.size();
157  const auto tRate = transProfile.GetRate();
158 
159  // check all contacts
160  for (size_t i_person1 = 0; i_person1 < pSize; i_person1++) {
161  // check if member is present today
162  const auto p1 = pMembers[i_person1];
163  if (p1->IsInPool(pType)) {
164  const double c_rate = GetContactRate(profile, p1, pSize);
165  // loop over possible contacts (contacts can be initiated by each member)
166  for (size_t i_person2 = 0; i_person2 < pSize; i_person2++) {
167  // check if not the same person
168  if (i_person1 != i_person2) {
169  // check if member is present today
170  const auto p2 = pMembers[i_person2];
171  if (p2->IsInPool(pType)) {
172  // check for contact
173  if (cHandler.HasContact(c_rate)) {
174  // log contact if person 1 is participating in survey
175  LP::Contact(cLogger, p1, p2, pType, simDay, c_rate, tRate);
176  // log contact if person 2 is participating in survey
177  LP::Contact(cLogger, p2, p1, pType, simDay, c_rate, tRate);
178 
179  // transmission & infection.
180  if (cHandler.HasTransmission(tRate)) {
181  auto& h1 = p1->GetHealth();
182  auto& h2 = p2->GetHealth();
183  // No secondary infections with TIC; just mark
184  // p2 as being recovered
185  if (h1.IsInfectious() && h2.IsSusceptible()) {
186  LP::Trans(cLogger, p1, p2, pType, simDay);
187  h2.StartInfection();
188  if (TIC)
189  h2.StopInfection();
190  } else if (h2.IsInfectious() && h1.IsSusceptible()) {
191  LP::Trans(cLogger, p2, p1, pType, simDay);
192  h1.StartInfection();
193  if (TIC)
194  h1.StopInfection();
195  }
196  }
197  }
198  }
199  }
200  }
201  }

209 template <ContactLogMode::Id LL, bool TIC>
210 void Infector<LL, TIC, true>::Exec(ContactPool& pool, const AgeContactProfile& profile,
211  const TransmissionProfile& transProfile, ContactHandler& cHandler,
212  unsigned short int simDay, shared_ptr<spdlog::logger> cLogger)
213 {
214  using LP = LOG_POLICY<LL>;
215 
216  // check for infected members and sort
217  bool infectious_cases;
218  size_t num_cases;
219  tie(infectious_cases, num_cases) = pool.SortMembers();
220 
221  if (!infectious_cases) {
222  return;
223  }
224 
225  // set up some stuff
226  const auto c_type = pool.m_pool_type;
227  const auto c_immune = pool.m_index_immune;
228  const auto& c_members = pool.m_members;
229  const auto c_size = c_members.size();
230  const auto t_rate = transProfile.GetRate();
231 
232  // match infectious and susceptible members, skip last part (immune members)
233  for (size_t i_infected = 0; i_infected < num_cases; i_infected++) {
234  // check if member is present today
235  const auto p1 = c_members[i_infected];
236  if (p1->IsInPool(c_type)) {
237  auto& h1 = p1->GetHealth();
238  if (h1.IsInfectious()) {
239  const double c_rate_p1 = GetContactRate(profile, p1, c_size);
240  // loop over possible susceptible contacts
241  for (size_t i_contact = num_cases; i_contact < c_immune; i_contact++) {
242  // check if member is present today
243  const auto p2 = c_members[i_contact];
244  if (p2->IsInPool(c_type)) {
245  const double c_rate_p2 = GetContactRate(profile, p2, c_size);
246  if (cHandler.HasContactAndTransmission(c_rate_p1, t_rate) ||
247  cHandler.HasContactAndTransmission(c_rate_p2, t_rate)) {
248  auto& h2 = p2->GetHealth();
249  if (h1.IsInfectious() && h2.IsSusceptible()) {
250  h2.StartInfection();
251  // No secondary infections with TIC; just mark
252  // p2 as being recovered
253  if (TIC)
254  h2.StopInfection();
255  LP::Trans(cLogger, p1, p2, c_type, simDay);
256  }
257  }
258  }
259  }
260  }
261  }

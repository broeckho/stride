44 shared_ptr<Population> DefaultPopBuilder::MakePersons(shared_ptr<Population> pop)
45 {
46  //------------------------------------------------
47  // Read persons from file.
48  //------------------------------------------------
49  const auto file_name = m_config_pt.get<string>("run.population_file");
50  const auto use_install_dirs = m_config_pt.get<bool>("run.use_install_dirs");
51  const auto file_path = (use_install_dirs) ? FileSys::GetDataDir() /= file_name : filesys::path(file_name);
52  if (!is_regular_file(file_path)) {
53  throw runtime_error(string(__func__) + "> Population file " + file_path.string() + " not present.");
54  }
55 
56  ifstream pop_file;
57  pop_file.open(file_path.string());
58  if (!pop_file.is_open()) {
59  throw runtime_error(string(__func__) + "> Error opening population file " + file_path.string());
60  }
61 
62  string line;
63  getline(pop_file, line); // step over file header
64  unsigned int person_id = 0U;
65 
66  while (getline(pop_file, line)) {
67  const auto values = Split(line, ",");
68  const auto age = FromString<unsigned int>(values[0]);
69  const auto household_id = FromString<unsigned int>(values[1]);
70  const auto school_id = FromString<unsigned int>(values[2]);
71  const auto work_id = FromString<unsigned int>(values[3]);
72  const auto primary_community_id = FromString<unsigned int>(values[4]);
73  const auto secondary_community_id = FromString<unsigned int>(values[5]);
74 
75  pop->CreatePerson(person_id, age, household_id, school_id, 0, work_id, primary_community_id,
76  secondary_community_id);
77  ++person_id;
78  }
79 
80  pop_file.close();
81  return pop;
82 }

84 shared_ptr<Population> DefaultPopBuilder::Build(shared_ptr<Population> pop)
85 {
86  //------------------------------------------------
87  // Check validity of input data.
88  //------------------------------------------------
89  const auto seeding_rate = m_config_pt.get<double>("run.seeding_rate");
90  if (seeding_rate > 1.0) {
91  throw runtime_error(string(__func__) + "> Bad input data for seeding_rate.");
92  }
93 
94  //------------------------------------------------
95  // Add persons
96  //------------------------------------------------
97  MakePersons(pop);
98 
99  // --------------------------------------------------------------
100  // Determine maximum pool ids in population.
101  // --------------------------------------------------------------
102  IdSubscriptArray<unsigned int> max_ids{0U};
103  for (const auto& p : *pop) {
104  for (Id typ : IdList) {
105  max_ids[typ] = max(max_ids[typ], p.GetPoolId(typ));
106  }
107  }
108  // --------------------------------------------------------------
109  // Initialize poolSys with empty ContactPools (even for Id=0).
110  // --------------------------------------------------------------
111  for (Id typ : IdList) {
112  for (unsigned int i = 1; i < max_ids[typ] + 1; i++) {
113  pop->m_pool_sys[typ].emplace_back(ContactPool(i, typ));
114  }
115  }
116 
117  // --------------------------------------------------------------
118  // Insert persons (pointers) in their contactpools. Having Id 0
119  // means "not belonging pool of that type" (e.g. school/ work -
120  // cannot belong to both, or e.g. out-of-work).
121  //
122  // Pools are uniquely identified by (typ, subscript) and a Person
123  // belongs, for typ, to pool with subscript p.GetPoolId(typ).
124  // Defensive measure: we have a pool for Id 0 and leave it empty.
125  // --------------------------------------------------------------
126  for (auto& p : *pop) {
127  for (Id typ : IdList) {
128  const auto poolId = p.GetPoolId(typ);
129  if (poolId > 0) {
130  pop->m_pool_sys[typ][poolId].AddMember(&p);
131  }
132  }
133  }
134 
135  return pop;
136 }

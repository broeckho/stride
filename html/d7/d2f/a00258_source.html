27 void K12SchoolGenerator::Apply(shared_ptr<GeoGrid> geoGrid, GeoGridConfig& geoGridConfig)
28 {
29  // 1. given the number of persons of school age, calculate number of schools; schools
30  // have 500 pupils on average
31  // 2. assign schools to a location by using a discrete distribution which reflects the
32  // relative number of pupils for that location; the relative number of pupils is set
33  // to the relative population w.r.t the total population.
34 
35  int pupilCount = geoGridConfig.calculated.compulsoryPupils;
36  auto schoolCount = static_cast<int>(ceil(pupilCount / geoGridConfig.constants.meanK12SchoolSize));
37 
38  vector<double> weights;
39  for (const shared_ptr<Location>& loc : *geoGrid) {
40  weights.push_back(loc->GetRelativePopulationSize());
41  }
42 
43  if (weights.empty()) {
44  // trng can't handle empty vectors
45  return;
46  }
47 
48  auto dist = m_rnManager[0].variate_generator(trng::discrete_dist(weights.begin(), weights.end()));
49 
50  for (int i = 0; i < schoolCount; i++) {
51  auto loc = (*geoGrid)[dist()];
52  auto k12 = make_shared<K12School>(geoGridConfig.generated.contactCenters++);
53  k12->Fill(geoGrid);
54  loc->AddContactCenter(k12);
55  }
56 }

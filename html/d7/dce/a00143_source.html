31 void CollegeGenerator::Apply(GeoGrid& geoGrid, const GeoGridConfig& geoGridConfig)
32 {
33  const auto studentCount = geoGridConfig.popInfo.popcount_college;
34  const auto collegeCount =
35  static_cast<unsigned int>(ceil(studentCount / static_cast<double>(geoGridConfig.pools.college_size)));
36  const auto cities = geoGrid.TopK(10);
37 
38  if (cities.empty()) {
39  // trng can't handle empty vectors
40  return;
41  }
42 
43  // Aggregate population in TopK cities.
44  auto totalPop = 0U;
45  for (const auto& c : cities) {
46  totalPop += c->GetPopCount();
47  }
48 
49  // Weights determined by relative population in city.
50  vector<double> weights;
51  for (const auto& c : cities) {
52  const auto weight = static_cast<double>(c->GetPopCount()) / static_cast<double>(totalPop);
53  AssertThrow(weight >= 0 && weight <= 1 && !std::isnan(weight),
54  "CollegeGenerator> Invalid weight: " + to_string(weight), m_logger);
55  weights.push_back(weight);
56  }
57 
58  const auto dist = m_rn_man.GetDiscreteGenerator(weights, 0U);
59  auto pop = geoGrid.GetPopulation();
60 
61  for (auto i = 0U; i < collegeCount; i++) {
62  auto loc = cities[dist()];
63  AddPools(*loc, pop, geoGridConfig.pools.pools_per_college);
64  }
65 }

43 void WorkplacePopulator::Apply(shared_ptr<GeoGrid> geoGrid, const GeoGridConfig& geoGridConfig)
44 {
45  m_logger->info("Starting to populate Workplaces");
46 
47  m_geoGrid = geoGrid;
48  m_geoGridConfig = geoGridConfig;
49  m_fractionCommutingStudents = 0;
50  m_workplacesInCity.clear();
51  m_currentLoc = nullptr;
52  m_assignedTo0 = 0;
53  m_assignedCommuting = 0;
54  m_assignedNotCommuting = 0;
55  m_distNonCommuting = DiscreteDistType();
56  m_nearByWorkplaces.clear();
57  m_disCommuting = DiscreteDistType();
58  m_commutingLocations.clear();
59 
60  CalculateFractionCommutingStudents();
61  CalculateWorkplacesInCity();
62 
63  // for every location
64  for (const auto& loc : *geoGrid) {
65  if (loc->GetPopCount() == 0) {
66  continue;
67  }
68  m_currentLoc = loc;
69  CalculateCommutingLocations();
70  CalculateNearbyWorkspaces();
71 
72  // 2. for everyone of working age: decide between work or college (iff of College age)
73  for (const auto& household : loc->GetContactCentersOfType<Household>()) {
74  auto contactPool = household->GetPools()[0];
75  for (auto p : *contactPool) {
76  if (AgeBrackets::Workplace::HasAge((p->GetAge()))) {
77  bool isStudent = MakeChoice(geoGridConfig.input.participation_college);
78  bool isActiveWorker = MakeChoice(geoGridConfig.input.particpation_workplace);
79 
80  if ((AgeBrackets::College::HasAge(p->GetAge()) && !isStudent) ||
81  isActiveWorker) {
82  AssignActive(p);
83  } else {
84  // this person has no employment
85  p->SetPoolId(Id::Workplace, 0);
86  m_assignedTo0++;
87  }
88  }
89  }
90  }
91  }
92 
93  m_logger->info("Populated workplaces, assigned to 0 {}, assigned (commuting) {} assigned (not commuting) {} ",
94  m_assignedTo0, m_assignedCommuting, m_assignedNotCommuting);
95 }

97 void WorkplacePopulator::CalculateFractionCommutingStudents()
98 {
99  m_fractionCommutingStudents = 0;
100  if (static_cast<bool>(m_geoGridConfig.input.fraction_workplace_commuters) &&
101  m_geoGridConfig.popInfo.popcount_workplace) {
102  m_fractionCommutingStudents =
103  (m_geoGridConfig.popInfo.popcount_college * m_geoGridConfig.input.fraction_college_commuters) /
104  (m_geoGridConfig.popInfo.popcount_workplace * m_geoGridConfig.input.fraction_workplace_commuters);
105  }

108 void WorkplacePopulator::CalculateWorkplacesInCity()
109 {
110  for (const shared_ptr<Location>& loc : *m_geoGrid) {
111  vector<ContactPool*> contactPools;
112  for (const auto& wp : loc->GetContactCentersOfType<Workplace>()) {
113  contactPools.insert(contactPools.end(), wp->begin(), wp->end());
114  }
115 
116  auto disPools = m_rnManager[0].variate_generator(
117  trng::uniform_int_dist(0, static_cast<trng::uniform_int_dist::result_type>(contactPools.size())));
118 
119  m_workplacesInCity[loc.get()] = {contactPools, disPools};
120  }
121 }

123 void WorkplacePopulator::AssignActive(Person* person)
124 {
125  // this person is (student and active) or active
126  if (!m_commutingLocations.empty() && MakeChoice(m_geoGridConfig.input.fraction_workplace_commuters)) {
127  // this person is commuting
128  const auto& info = m_workplacesInCity[m_commutingLocations[m_disCommuting()]];
129  const auto id = info.second(); // id of the location this person is commuting to
130  info.first[id]->AddMember(person);
131  person->SetPoolId(Id::Workplace, info.first[id]->GetId());
132  m_assignedCommuting++;
133  } else {
134  // this person is not commuting
135  const auto id = m_distNonCommuting();
136  m_nearByWorkplaces[id]->AddMember(person);
137  person->SetPoolId(Id::Workplace, m_nearByWorkplaces[id]->GetId());
138  m_assignedNotCommuting++;
139  }
140 }

142 void WorkplacePopulator::CalculateCommutingLocations()
143 {
144  // find all Workplaces were employees from this location commute to
145  m_commutingLocations.clear();
146  m_disCommuting = DiscreteDistType();
147 
148  vector<double> commutingWeights;
149  for (const pair<Location*, double>& commute : m_currentLoc->GetOutgoingCommutingCities()) {
150  const auto& workplaces = commute.first->GetContactCentersOfType<Workplace>();
151  if (!workplaces.empty()) {
152  m_commutingLocations.push_back(commute.first);
153  const auto weight = commute.second - (commute.second * m_fractionCommutingStudents);
154  commutingWeights.push_back(weight);
155  AssertThrow(weight >= 0.0 && weight <= 1.0 && !isnan(weight),
156  "Invalid weight: " + to_string(weight), m_logger);
157  }
158  }
159 
160  if (!commutingWeights.empty()) {
161  m_disCommuting = m_rnManager[0].variate_generator(
162  trng::discrete_dist(commutingWeights.begin(), commutingWeights.end()));
163  }
164 }

31 void WorkplaceGenerator::Apply(shared_ptr<GeoGrid> geoGrid, const GeoGridConfig& geoGridConfig,
32  unsigned int& contactCenterCounter)
33 {
34  // 1. active people count and the commuting people count are given
35  // 2. count the workplaces, each workplace has an average of 20 employees
36  // 3. count the working people at each location = #residents + #incoming commuters - #outgoing commuters
37  // 4. use the last information for the distribution
38  // 5. assign each workplace to a location
39 
40  const auto EmployeeCount = geoGridConfig.popInfo.popcount_workplace;
41  const auto WorkplacesCount =
42  static_cast<unsigned int>(ceil(EmployeeCount / static_cast<double>(geoGridConfig.pools.workplace_size)));
43 
44  // = for each location #residents + #incoming commuting people - #outgoing commuting people
45  vector<double> weights;
46  for (const auto& loc : *geoGrid) {
47  const double ActivePeopleCount =
48  (loc->GetPopCount() +
49  loc->GetIncomingCommuterCount(geoGridConfig.input.fraction_workplace_commuters) -
50  loc->GetOutgoingCommuterCount(geoGridConfig.input.fraction_workplace_commuters) *
51  geoGridConfig.input.particpation_workplace);
52 
53  const double weight = ActivePeopleCount / EmployeeCount;
54  AssertThrow(weight >= 0 && weight <= 1 && !std::isnan(weight), "Invalid weight: " + to_string(weight),
55  m_logger);
56  weights.push_back(weight);
57  }
58 
59  if (weights.empty()) {
60  // trng can't handle empty vectors
61  return;
62  }
63 
64  const auto dist = m_rnManager[0].variate_generator(trng::discrete_dist(weights.begin(), weights.end()));
65 
66  for (auto i = 0U; i < WorkplacesCount; i++) {
67  const auto loc = (*geoGrid)[dist()];
68  const auto w = make_shared<Workplace>(contactCenterCounter++);
69  w->Fill(geoGridConfig, geoGrid);
70  loc->AddContactCenter(w);
71  }
72 }

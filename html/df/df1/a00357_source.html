28 void PrimaryCommunityPopulator::Apply(std::shared_ptr<GeoGrid> geoGrid, GeoGridConfig&)
29 {
30  m_logger->info("Starting to populate Primary Communities");
31 
32  std::set<stride::ContactPool*> found;
33  // for every location
34  for (const std::shared_ptr<Location>& loc : *geoGrid) {
35  if (loc->GetPopulation() == 0) {
36  continue;
37  }
38 
39  // 1. find all communities in an area of 10-k*10 km
40  const auto& community_pools = GetContactPoolInIncreasingRadius<PrimaryCommunity>(geoGrid, loc);
41 
42  // 2. for every household assign a community
43  const auto dist = m_rnManager[0].variate_generator(trng::uniform_int_dist(
44  0, static_cast<trng::uniform_int_dist::result_type>(community_pools.size())));
45 
46  for (const std::shared_ptr<ContactCenter>& household : loc->GetContactCentersOfType<Household>()) {
47  stride::ContactPool* contactPool = household->GetPools()[0];
48  for (stride::Person* person : *contactPool) {
49  const auto pool = dist();
50  found.insert(community_pools[pool]);
51  community_pools[pool]->AddMember(person);
52  person->SetPrimaryCommunityId(community_pools[pool]->GetId());
53  }
54  }
55  }
56 
57  m_logger->info("Finished populating Primary Communities");
58  m_logger->info("Used {} different Primary communities", found.size());
59 }

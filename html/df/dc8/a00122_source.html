88 void CSV::ReadFromStream(istream& inputStream)
89 {
90  string line;
91 
92  // process header, get labels and columnCount
93  getline(inputStream, line);
94  line = Trim(line);
95  vector<string> headerLabels = Split(line, ",");
96  for (const string& label : headerLabels) {
97  m_labels.push_back(Trim(label, "\""));
98  }
99  m_column_count = m_labels.size();
100 
101  // body
102  while (getline(inputStream, line)) {
103  line = Trim(line);
104  if (!line.empty()) {
105  vector<string> values = Split(line, ",");
106  AddRow(values.begin(), values.end());
107  }
108  }
109 }

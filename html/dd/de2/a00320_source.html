40 void WorkplacePopulator::Apply(GeoGrid& geoGrid, const GeoGridConfig& geoGridConfig)
41 {
42  m_logger->trace("Starting to populate Workplaces");
43 
44  m_geogrid_config = geoGridConfig;
45  m_gen_commute = function<int()>();
46  m_gen_non_commute = function<int()>();
47 
48  m_nearby_wp.clear();
49  m_commuting_locations.clear();
50 
51  const auto fractionCommutingStudents = FractionCommutingStudents();
52 
53  // for every location
54  for (const auto& loc : geoGrid) {
55  if (loc->GetPopCount() == 0) {
56  continue;
57  }
58  CommutingLocations(loc, fractionCommutingStudents);
59  NearbyWorkspacePools(geoGrid, loc);
60 
61  // 2. for everyone of working age: decide between work or college (iff of College age)
62  for (auto& hhPool : loc->RefPools(Id::Household)) {
63  for (auto p : *hhPool) {
64  if (AgeBrackets::Workplace::HasAge((p->GetAge()))) {
65  bool isStudent = MakeChoice(geoGridConfig.input.participation_college);
66  bool isActiveWorker = MakeChoice(geoGridConfig.input.particpation_workplace);
67 
68  if ((AgeBrackets::College::HasAge(p->GetAge()) && !isStudent) ||
69  isActiveWorker) {
70  AssignActive(p);
71  } else {
72  // this person has no employment
73  p->SetPoolId(Id::Workplace, 0);
74  }
75  }
76  }
77  }
78  }
79 
80  m_logger->trace("Done populating Workplaces");
81 }

83 void WorkplacePopulator::AssignActive(Person* person)
84 {
85  // this person is (student and active) or active
86  if (!m_commuting_locations.empty() && MakeChoice(m_geogrid_config.input.fraction_workplace_commuters)) {
87  // this person commutes to the Location
88  auto loc = m_commuting_locations[m_gen_commute()];
89  // and in particular to pool
90  auto& pools = loc->RefPools(Id::Workplace);
91  auto gen = m_rn_man.GetUniformIntGenerator(0, static_cast<int>(pools.size()), 0U);
92  auto p = pools[gen()];
93  // so that's it
94  p->AddMember(person);
95  person->SetPoolId(Id::Workplace, p->GetId());
96  } else {
97  // this person does not
98  const auto idraw = m_gen_non_commute();
99  m_nearby_wp[idraw]->AddMember(person);
100  person->SetPoolId(Id::Workplace, m_nearby_wp[idraw]->GetId());
101  }
102 }

104 void WorkplacePopulator::CommutingLocations(const std::shared_ptr<Location>& loc, double fractionCommuteStudents)
105 {
106  // find all Workplaces were employees from this location commute to
107  m_commuting_locations.clear();
108  m_gen_commute = function<int()>();
109 
110  vector<double> commutingWeights;
111  for (const pair<Location*, double>& commute : loc->CRefOutgoingCommutes()) {
112  const auto& workplaces = commute.first->RefPools(Id::Workplace);
113  if (!workplaces.empty()) {
114  m_commuting_locations.push_back(commute.first);
115  const auto weight = commute.second - (commute.second * fractionCommuteStudents);
116  commutingWeights.push_back(weight);
117  AssertThrow(weight >= 0.0 && weight <= 1.0 && !isnan(weight),
118  "Invalid weight: " + to_string(weight), m_logger);
119  }
120  }
121 
122  if (!commutingWeights.empty()) {
123  m_gen_commute = m_rn_man.GetDiscreteGenerator(commutingWeights, 0U);
124  }
125 }

127 double WorkplacePopulator::FractionCommutingStudents()
128 {
129  double fraction = 0.0;
130  if (static_cast<bool>(m_geogrid_config.input.fraction_workplace_commuters) &&
131  m_geogrid_config.popInfo.popcount_workplace) {
132  fraction =
133  (m_geogrid_config.popInfo.popcount_college * m_geogrid_config.input.fraction_college_commuters) /
134  (m_geogrid_config.popInfo.popcount_workplace * m_geogrid_config.input.fraction_workplace_commuters);
135  }
136  return fraction;
137 }

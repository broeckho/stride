40 void DiseaseSeeder::Seed(std::shared_ptr<Population> pop)
41 {
42  // --------------------------------------------------------------
43  // Population immunity (natural immunity & vaccination).
44  // --------------------------------------------------------------
45  const auto immunityProfile = m_config.get<std::string>("run.immunity_profile");
46  Vaccinate("immunity", immunityProfile, pop->CRefPoolSys().CRefPools<Id::Household>());
47 
48  const auto vaccinationProfile = m_config.get<std::string>("run.vaccine_profile");
49  Vaccinate("vaccine", vaccinationProfile, pop->CRefPoolSys().CRefPools<Id::Household>());
50 
51  // --------------------------------------------------------------
52  // Seed infected persons.
53  // --------------------------------------------------------------
54  const auto sRate = m_config.get<double>("run.seeding_rate");
55  const auto sAgeMin = m_config.get<double>("run.seeding_age_min", 1);
56  const auto sAgeMax = m_config.get<double>("run.seeding_age_max", 99);
57  const auto popSize = pop->size();
58  const auto maxPopIndex = static_cast<int>(popSize - 1);
59  auto generator = m_rn_man.GetUniformIntGenerator(0, maxPopIndex, 0U);
60  auto& logger = pop->RefContactLogger();
61  const string log_level = m_config.get<string>("run.contact_log_level", "None");
62 
63  auto numInfected = static_cast<unsigned int>(floor(static_cast<double>(popSize) * sRate));
64  while (numInfected > 0) {
65  Person& p = pop->at(static_cast<size_t>(generator()));
66  if (p.GetHealth().IsSusceptible() && (p.GetAge() >= sAgeMin) && (p.GetAge() <= sAgeMax)) {
67  p.GetHealth().StartInfection();
68  numInfected--;
69  if (log_level != "None") {
70  logger->info("[PRIM] {} {} {} {} {} {}", p.GetId(), -1, p.GetAge(), -1, -1, -1);
71  }
72  }
73  }
74 }

76 void DiseaseSeeder::Vaccinate(const std::string& immunityType, const std::string& immunizationProfile,
77  const SegmentedVector<ContactPool>& immunityPools)
78 {
79  std::vector<double> immunityDistribution;
80  double linkProbability = 0;
81  Immunizer immunizer(m_rn_man);
82 
83  if (immunizationProfile == "Random") {
84  const auto immunityRate = m_config.get<double>("run." + ToLower(immunityType) + "_rate");
85  for (unsigned int index_age = 0; index_age < 100; index_age++) {
86  immunityDistribution.push_back(immunityRate);
87  }
88  immunizer.Random(immunityPools, immunityDistribution, linkProbability);
89  } else if (immunizationProfile == "AgeDependent") {
90  const auto immunityFile = m_config.get<string>("run." + ToLower(immunityType) + "_distribution_file");
91  const ptree& immunity_pt = FileSys::ReadPtreeFile(immunityFile);
92 
93  linkProbability = m_config.get<double>("run." + ToLower(immunityType) + "_link_probability");
94 
95  for (unsigned int index_age = 0; index_age < 100; index_age++) {
96  auto immunityRate = immunity_pt.get<double>("immunity.age" + std::to_string(index_age));
97  immunityDistribution.push_back(immunityRate);
98  }
99  immunizer.Random(immunityPools, immunityDistribution, linkProbability);
100 
101  } else if (immunizationProfile == "Cocoon") {
102  immunizer.Cocoon(immunityPools, immunityDistribution, linkProbability);
103  }
104 }

114 vector<shared_ptr<Location>> GeoGrid::TopK(size_t k) const
115 {
116  auto cmp = [](const shared_ptr<Location>& rhs, const shared_ptr<Location>& lhs) {
117  return rhs->GetPopCount() > lhs->GetPopCount();
118  };
119 
120  priority_queue<shared_ptr<Location>, vector<shared_ptr<Location>>, decltype(cmp)> queue(cmp);
121  for (const auto& loc : m_locations) {
122  queue.push(loc);
123  if (queue.size() > k) {
124  queue.pop();
125  }
126  }
127 
128  vector<shared_ptr<Location>> topLocations;
129  while (!queue.empty()) {
130  auto loc = queue.top();
131  topLocations.push_back(loc);
132  queue.pop();
133  }
134 
135  return topLocations;
136 }

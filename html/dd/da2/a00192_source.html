62 void Sim::TimeStep()
63 {
64  // Logic where you compute (on the basis of input/config for initial day or on the basis of
65  // number of sick persons, duration of epidemic etc) what kind of DaysOff scheme you apply.
66  const auto daysOff = std::make_shared<DaysOffStandard>(m_calendar);
67  const bool isWorkOff = daysOff->IsWorkOff();
68  const bool isSchoolOff = daysOff->IsSchoolOff();
69 
70  // To be used in update of population & contact pools.
71  Population& population = *m_population;
72  auto& poolSys = population.GetContactPoolSys();
73  auto contactLogger = population.GetContactLogger();
74  const auto simDay = m_calendar->GetSimulationDay();
75  const auto& infector = *m_infector;
76 
77 #pragma omp parallel num_threads(m_num_threads)
78  {
79  // Update health status and presence/absence in pools
80  // depending on health status, work/school day.
81 #pragma omp for schedule(static)
82  for (size_t i = 0; i < population.size(); ++i) {
83  population[i].Update(isWorkOff, isSchoolOff);
84  }
85 
86  // Infector updates individuals for contacts & transmission within each pool.
87  // Skip pools with id = 0, because it means Not Applicable.
88  const auto thread_num = static_cast<unsigned int>(omp_get_thread_num());
89  for (auto typ : ContactPoolType::IdList) {
90  if ((typ == ContactPoolType::Id::Work && isWorkOff) ||
91  (typ == ContactPoolType::Id::School && isSchoolOff)) {
92  continue;
93  }
94 #pragma omp for schedule(static)
95  for (size_t i = 1; i < poolSys[typ].size(); i++) { // NOLINT
96  infector(poolSys[typ][i], m_contact_profiles[typ], m_transmission_profile,
97  m_handlers[thread_num], simDay, contactLogger);
98  }
99  }
100  }
101 
102  m_population->GetContactLogger()->flush();
103  m_calendar->AdvanceDay();
104 }

27 void CommunityGenerator::Apply(std::shared_ptr<GeoGrid> geoGrid, GeoGridConfig& geoGridConfig)
28 {
29  /*
30  * 1. calculate amount of communities, each community has average 2000 persons
31  * 2. assign communities to a location by using a discrete distribution which reflects the relative amount of
32  * people for that location
33  */
34 
35  int populationSize = geoGridConfig.input.populationSize;
36  auto amountOfCommunities =
37  static_cast<int>(std::ceil(populationSize / geoGridConfig.constants.meanCommunitySize));
38 
39  std::vector<double> weights;
40 
41  for (const std::shared_ptr<Location>& loc : *geoGrid) {
42  double weight =
43  static_cast<double>(loc->GetPopulation()) / static_cast<double>(geoGridConfig.input.populationSize);
44 
45  CheckWeight("CommunityGenerator", weight);
46 
47  weights.push_back(weight);
48  }
49 
50  if (weights.empty()) {
51  // trng can't handle empty vectors
52  return;
53  }
54 
55  auto dist = m_rnManager[0].variate_generator(trng::discrete_dist(weights.begin(), weights.end()));
56  for (int communityId = 0; communityId < amountOfCommunities; communityId++) {
57  int locationId = dist();
58  std::shared_ptr<Location> loc = (*geoGrid)[locationId];
59  std::shared_ptr<PrimaryCommunity> community = std::make_shared<PrimaryCommunity>(geoGridConfig.generated.contactCenters++);
60  community->Fill(geoGrid);
61  loc->AddContactCenter(community);
62  }
63  for (int communityId = 0; communityId < amountOfCommunities; communityId++) {
64  int locationId = dist();
65  std::shared_ptr<Location> loc = (*geoGrid)[locationId];
66  std::shared_ptr<SecondaryCommunity> community = std::make_shared<SecondaryCommunity>(geoGridConfig.generated.contactCenters++);
67  community->Fill(geoGrid);
68  loc->AddContactCenter(community);
69  }
70 }

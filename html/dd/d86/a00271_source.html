46 vector<ContactPool*> Populator::GetNearbyPools(Id id, const GeoGrid& geoGrid, const Location& start,
47  double startRadius) const
48 {
49  double currentRadius = startRadius;
50  vector<ContactPool*> pools;
51 
52  while (pools.empty()) {
53  for (const Location* nearLoc : geoGrid.LocationsInRadius(start, currentRadius)) {
54  const auto& centers = nearLoc->CRefCenters(id);
55  for (const auto& center : centers) {
56  pools.insert(pools.end(), center->begin(), center->end());
57  }
58  }
59  currentRadius *= 2;
60  if (currentRadius == numeric_limits<double>::infinity()) {
61  break;
62  }
63  }
64  return pools;
65 }
